CLAIM c1: the standard Cauchy density $1/(\pi(1+x^2))$ integrates to 1 over the real line
```python
from sympy import *

x = symbols('x', real=True)
f = 1/(pi*(1 + x**2))
assert simplify(integrate(f, (x, -oo, oo)) - 1) == 0
```

CLAIM c2: the location-scale Cauchy density with location mu0 and scale gamma0 > 0 integrates to 1
```python
from sympy import *

theta = symbols('theta', real=True)
mu0 = symbols('mu0', real=True)
g0 = symbols('gamma0', positive=True)

f = 1/(g0*pi*(1 + ((theta - mu0)/g0)**2))
assert simplify(integrate(f, (theta, -oo, oo)) - 1) == 0
```

CLAIM c4: phi = sigma^2 with sigma half-Cauchy has density 1/(pi*sqrt(phi)*(1+phi)) on (0, oo), and it is proper
```python
from sympy import *

s, phi = symbols('sigma phi', positive=True)

# change of variables: pi(phi) = pi(sigma(phi)) * |d sigma / d phi|
pi_sigma = 2/(pi*(1 + s**2))
sigma_of_phi = sqrt(phi)
jac = diff(sigma_of_phi, phi)                      # 1/(2*sqrt(phi)) > 0

derived = simplify(pi_sigma.subs(s, sigma_of_phi) * jac)
stated = 1/(pi*sqrt(phi)*(1 + phi))

assert simplify(derived - stated) == 0
assert simplify(integrate(stated, (phi, 0, oo)) - 1) == 0
```

CLAIM c5: alpha = 1/(1+phi) has density 1/(pi*sqrt(alpha*(1-alpha))) = Beta(1/2,1/2) density, and B(1/2,1/2) = pi
```python
from sympy import *

phi = symbols('phi', positive=True)
a = symbols('alpha', positive=True)

pi_phi = 1/(pi*sqrt(phi)*(1 + phi))

# phi = (1-alpha)/alpha  on  0 < alpha < 1;  |d phi / d alpha| = 1/alpha^2
phi_of_a = (1 - a)/a
jac = 1/a**2

derived = pi_phi.subs(phi, phi_of_a) * jac
stated = 1/(pi*sqrt(a*(1 - a)))

# verify equality on 0 < alpha < 1 by exact rational substitution
for r in [Rational(1, 4), Rational(1, 2), Rational(9, 10), Rational(1, 100)]:
    assert simplify(derived.subs(a, r) - stated.subs(a, r)) == 0

# stated density is exactly the Beta(1/2, 1/2) density:
# B(1/2,1/2) = Gamma(1/2)^2/Gamma(1) = pi
B = gamma(Rational(1, 2))**2/gamma(1)
assert simplify(B - pi) == 0
assert simplify(beta(Rational(1, 2), Rational(1, 2)).rewrite(gamma) - pi) == 0

for r in [Rational(1, 4), Rational(1, 2), Rational(9, 10)]:
    beta_pdf_r = r**(Rational(1, 2) - 1)*(1 - r)**(Rational(1, 2) - 1)/B
    assert simplify(stated.subs(a, r) - beta_pdf_r) == 0

# proper on (0,1)
assert simplify(integrate(stated, (a, 0, 1)) - 1) == 0
```
