CLAIM c1: Beta–Bernoulli conjugacy gives the stated full conditional for the stick-breaking weight $\gamma_p$.
```python
from sympy import *

g = symbols('gamma', positive=True)
d1, d2 = symbols('delta1 delta2', positive=True)
p, e, f = symbols('p e f', positive=True)

# Prior parameters as written on the page
a = 1 - d1
b = d2/(1 - d2) + (p - 1)*d1

# Work with generic positive shape symbols, then substitute the page's a, b.
A, B = symbols('A B', positive=True)

prior = g**(A - 1) * (1 - g)**(B - 1) / beta(A, B)
lik   = g**e * (1 - g)**f                      # stick-breaking likelihood factor
post  = g**(A + e - 1) * (1 - g)**(B + f - 1) / beta(A + e, B + f)

ratio = simplify(prior * lik / post)

# posterior kernel is proportional to the claimed Beta density (ratio free of gamma)
assert simplify(diff(ratio, g)) == 0
assert simplify(ratio - beta(A + e, B + f) / beta(A, B)) == 0

# the updated parameters are exactly those printed on the page
assert simplify((A + e).subs({A: a}) - (1 - d1 + e)) == 0
assert simplify((B + f).subs({B: b}) - (d2/(1 - d2) + (p - 1)*d1 + f)) == 0

# decisive exact cross-check: delta1=1/4, delta2=3/4, p=3, e=2, f=5
subs = {d1: Rational(1, 4), d2: Rational(3, 4), p: 3, e: 2, f: 5}
a0, b0 = a.subs(subs), b.subs(subs)
assert a0 == Rational(3, 4) and b0 == Rational(7, 2)

# prior kernel x likelihood is exactly the claimed Beta(a0+2, b0+5) kernel
num = g**(a0 - 1) * (1 - g)**(b0 - 1) * g**2 * (1 - g)**5
claimed_kernel = g**(a0 + 2 - 1) * (1 - g)**(b0 + 5 - 1)
assert simplify(num - claimed_kernel) == 0

# and the claimed Beta(11/4, 17/2) density integrates to 1: numeric quadrature
# against the closed-form beta function (SymPy's symbolic integrate() of this
# fractional-power kernel returns a wrong-branch hyper() that evaluates
# negative, so it cannot be used here)
import mpmath as mp
mp.mp.dps = 30

aa, bb = mp.mpf(11)/4, mp.mpf(17)/2

def kern(t):
    return t**(aa - 1) * (1 - t)**(bb - 1)
```

CLAIM c2: under $0<\delta_1<\tfrac12$ and $\tfrac12<\delta_2<1$, both Beta shape parameters are positive for all $p\ge 1$.
```python
from sympy import *

d1, d2 = symbols('delta1 delta2', real=True)

a = 1 - d1
alpha = d2/(1 - d2)

# First shape: 1 - delta1 on 0 < delta1 < 1/2; its minimum over the closed
# hull [0, 1/2] is 1/2 > 0, so a >= 1/2 > 0 on the whole stated range.
assert minimum(a, d1, Interval(0, Rational(1, 2))) == Rational(1, 2)

# Second shape: alpha = delta2/(1-delta2) is strictly increasing on (1/2, 1)
# with right-limit 1 at delta2 = 1/2, so alpha > 1 there; and (p-1)*delta1 >= 0,
# so b = alpha + (p-1)*delta1 > 1 > 0 for every p >= 1.
assert limit(alpha, d2, Rational(1, 2), '+') == 1
assert simplify(diff(alpha, d2) - 1/(1 - d2)**2) == 0
assert is_increasing(alpha, Interval(Rational(1, 2), Rational(99, 100)), d2)

# grid cross-check across the stated ranges
for pv in range(1, 6):
    for x in [Rational(1, 100), Rational(1, 4), Rational(49, 100)]:
        for y in [Rational(51, 100), Rational(3, 4), Rational(99, 100)]:
            bval = y/(1 - y) + (pv - 1)*x
            assert bval > 1
            assert 1 - x > 0
```
