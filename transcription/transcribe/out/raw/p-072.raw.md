**Definition.** A $100(1-\alpha)\%$ credible interval $(a,b)$ is a *H.P.D. interval* if for any $\theta_1 \in (a,b)$ and $\theta_2 \notin (a,b)$

$$p(\theta_1|y) \geq p(\theta_2|y).$$

$$\int_a^b p(\theta|y)d\theta = 1-\alpha \tag{8.1}$$

$$p(a|y) = p(b|y) \tag{8.2}$$

CLAIM c1: On the unimodal Beta(2,3) posterior with 1−α=0.9, the (a,b) solving ∫_a^b p = 0.9 and p(a)=p(b) satisfies the HPD property (min density inside ≥ max density outside), equals the level set {θ: p(θ) ≥ p(a)}, and is shorter than the equal-tailed 90% interval.
```python
import numpy as np
from scipy.stats import beta as Beta
from scipy.optimize import brentq

A, B = 2.0, 3.0
rv = Beta(A, B)
pdf, cdf, ppf = rv.pdf, rv.cdf, rv.ppf
mode = (A - 1) / (A + B - 2)          # 1/3
level = 0.90
pmax = pdf(mode)

# Parametrize by the horizontal cut C: the level set {pdf >= C} is [a(C), b(C)].
# This avoids root-bracketing failures near the mode.
def endpoints(C):
    a = brentq(lambda x: pdf(x) - C, 1e-12, mode, xtol=1e-14)
    b = brentq(lambda x: pdf(x) - C, mode, 1 - 1e-12, xtol=1e-14)
    return a, b

def coverage(C):
    a, b = endpoints(C)
    return cdf(b) - cdf(a)

# coverage(C) decreases continuously from ~1 to ~0 as C rises from 0 to pdf(mode)
C = brentq(lambda c: coverage(c) - level, 1e-9, pmax - 1e-9, xtol=1e-13)
a, b = endpoints(C)

# (8.1) and (8.2) hold at the solution
assert abs(cdf(b) - cdf(a) - level) < 1e-8
assert abs(pdf(a) - pdf(b)) < 1e-8

# HPD defining property: density inside >= density outside
inside = np.linspace(a, b, 20001)
outside = np.concatenate([np.linspace(1e-9, a, 10001), np.linspace(b, 1 - 1e-9, 10001)])
assert pdf(inside).min() >= C - 1e-8
assert pdf(outside).max() <= C + 1e-8

# equivalence with the level set {theta : p(theta) >= C_alpha}
grid = np.linspace(1e-9, 1 - 1e-9, 200001)
in_interval = (grid >= a) & (grid <= b)
in_levelset = pdf(grid) >= C - 1e-10
assert np.abs(in_interval.astype(int) - in_levelset.astype(int)).sum() <= 2  # boundary grid cells only

# HPD is shorter than the equal-tailed interval of the same coverage
et_a, et_b = ppf(0.05), ppf(0.95)
assert (b - a) < (et_b - et_a) - 1e-6
```

CLAIM c2: The 90% HPD interval for Beta(2,3) contains the mode (Remark (1)).
```python
from scipy.stats import beta as Beta
from scipy.optimize import brentq

A, B = 2.0, 3.0
rv = Beta(A, B)
pdf, cdf = rv.pdf, rv.cdf
mode = (A - 1) / (A + B - 2)
pmax = pdf(mode)

# Parametrize by the horizontal cut C (robust near the mode): {pdf >= C} = [a, b]
def endpoints(C):
    a = brentq(lambda x: pdf(x) - C, 1e-12, mode, xtol=1e-14)
    b = brentq(lambda x: pdf(x) - C, mode, 1 - 1e-12, xtol=1e-14)
    return a, b

C = brentq(lambda c: cdf(endpoints(c)[1]) - cdf(endpoints(c)[0]) - 0.90,
           1e-9, pmax - 1e-9, xtol=1e-13)
a, b = endpoints(C)

assert a < mode < b
```
