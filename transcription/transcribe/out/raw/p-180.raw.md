CLAIM c1: For $\phi \sim \mathrm{N}_d(0,M)$ with $M$ diagonal and positive, the joint density equals $\prod_j \mathrm{N}(\phi_j \mid 0, M_{jj})$, i.e. the components are independent with $\phi_j \sim \mathrm{N}(0,M_{jj})$.
```python
from sympy import *

d = 3
phis = symbols('phi1 phi2 phi3', real=True)
Ms = symbols('M11 M22 M33', positive=True)

M = diag(*Ms)
v = Matrix(phis)

# Multivariate normal N_d(0, M) density
joint = (2*pi)**Rational(-d, 2) * M.det()**Rational(-1, 2) * \
        exp(-Rational(1, 2) * (v.T * M.inv() * v)[0, 0])

# Product of univariate N(0, M_jj) densities
prod_marginals = 1
for p, m in zip(phis, Ms):
    prod_marginals *= 1/sqrt(2*pi*m) * exp(-p**2/(2*m))

assert simplify(joint - prod_marginals) == 0

# Each factor is exactly the N(0, M_jj) density: mean 0, variance M_jj
p1, m1 = phis[0], Ms[0]
dens = 1/sqrt(2*pi*m1) * exp(-p1**2/(2*m1))
assert simplify(integrate(dens, (p1, -oo, oo)) - 1) == 0
assert simplify(integrate(p1*dens, (p1, -oo, oo)) - 0) == 0
assert simplify(integrate(p1**2*dens, (p1, -oo, oo)) - m1) == 0
```
