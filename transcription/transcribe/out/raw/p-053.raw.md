===CLAIMS===

CLAIM c1: The kernel ratio $\pi/\pi_a$ equals $1/(1+\theta)$, is bounded by 1 on $\theta>0$, and its supremum is 1.
```python
from sympy import *

theta = Symbol('theta', positive=True)
a, b = symbols('a b', positive=True)

pi_kernel   = theta**(a-1) * exp(-b*theta) / (1 + theta)
pi_a_kernel = theta**(a-1) * exp(-b*theta)

# the booklet's stated ratio
ratio = simplify(pi_kernel / pi_a_kernel)
assert simplify(ratio - 1/(1 + theta)) == 0

# bound: 1/(1+theta) <= 1 for theta > 0, i.e. 1 - 1/(1+theta) = theta/(1+theta) >= 0
gap = simplify(1 - 1/(1 + theta))
assert simplify(gap - theta/(1 + theta)) == 0
assert gap.is_nonnegative is True

# the bound is tight: sup over theta > 0 is 1, so M = 1 and the accept rule U <= 1/(1+theta)
assert limit(1/(1 + theta), theta, 0, '+') == 1

# monotone decreasing => supremum attained only in the limit theta -> 0+
assert simplify(diff(1/(1 + theta), theta) + 1/(1 + theta)**2) == 0
assert (diff(1/(1 + theta), theta)).is_negative is True

# consistency: with M = 1, (1/M) * pi/pi_a is exactly the printed acceptance threshold
M = Integer(1)
assert simplify(Rational(1, 1)/M * ratio - 1/(1 + theta)) == 0
```

CLAIM c2: The number of trials to acceptance is geometric with success probability $1/M$: the pmf sums to 1, $P(N=1) = 1/M = P(\text{stops})$, and $E[N] = M$.
```python
from sympy import *

n = Symbol('n', positive=True, integer=True)

# check for several exact M > 1 (M is the envelope constant, necessarily >= 1)
for M in [Integer(2), Rational(5, 2), Integer(7), Rational(13, 4)]:
    p = Rational(1, 1) / M          # success (acceptance) probability = 1/M

    pmf = (1 - p)**(n - 1) * p

    # proper distribution
    assert simplify(summation(pmf, (n, 1, oo)) - 1) == 0

    # P(stops) on any given trial = 1/M, matching Note (1)
    assert simplify(pmf.subs(n, 1) - 1/M) == 0

    # expected number of trials to get one y equals M
    assert simplify(summation(n * pmf, (n, 1, oo)) - M) == 0
```
