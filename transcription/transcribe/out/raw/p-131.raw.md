CLAIM c1: The jumping rate in the tuning table is strictly decreasing in $k$.
```python
from fractions import Fraction as F

rows = [("5", "0.011"), ("3", "0.028"), ("2.5", "0.058"), ("2.0", "0.115"),
        ("1.8", "0.151"), ("1.6", "0.195"), ("1.5", "0.251"), ("1.4", "0.314"),
        ("1.3", "0.399"), ("1.25", "0.493"), ("1.18", "0.600"), ("1.11", "0.717"),
        ("1.00", "0.914")]
ks = [F(k) for k, _ in rows]
jr = [F(j) for _, j in rows]

# k is listed strictly decreasing down the table
assert all(ks[i] > ks[i + 1] for i in range(len(ks) - 1))
# jumping rate is strictly increasing down the table, i.e. strictly decreasing in k
assert all(jr[i] < jr[i + 1] for i in range(len(jr) - 1))
# equivalently: sorting by k ascending gives jumping rate descending
pairs = sorted(zip(ks, jr))
assert all(pairs[i][1] > pairs[i + 1][1] for i in range(len(pairs) - 1))
```
