CLAIM c1: The 100 equal-width bins ((i-1)/100, i/100) on (0,1) have middle points 0.005, 0.015, ..., 0.995.
```python
from sympy import Rational, simplify, symbols

i = symbols('i', positive=True, integer=True)
lo = (i - 1) / Rational(100)
hi = i / Rational(100)
mid = (lo + hi) / 2

assert simplify(mid - (2*i - 1) / Rational(200)) == 0

assert mid.subs(i, 1) == Rational(5, 1000)      # 0.005
assert mid.subs(i, 2) == Rational(15, 1000)     # 0.015
assert mid.subs(i, 100) == Rational(995, 1000)  # 0.995

# bins tile (0,1) exactly
assert lo.subs(i, 1) == 0 and hi.subs(i, 100) == 1
assert simplify((hi - lo) - Rational(1, 100)) == 0
```

CLAIM c2: With p_i = h_i / sum_j h_j, the cumulative sum p_1 + ... + p_100 equals 1.
```python
from sympy import symbols, together, fraction, expand
from fractions import Fraction

# symbolic: p_1 + ... + p_100 == 1 exactly (common denominator sum_j h_j)
h = symbols('h1:101', positive=True)
S = sum(h)
num, den = fraction(together(sum(hi / S for hi in h)))
assert expand(num - den) == 0

# numeric cross-check with exact rationals: heights 1..100
hv = [Fraction(k) for k in range(1, 101)]
Sv = sum(hv)
pv = [x / Sv for x in hv]
assert sum(pv) == 1

# CDF is nondecreasing, starts at p_1, and its last entry is exactly 1
cdf = []
acc = Fraction(0)
for x in pv:
    acc += x
    cdf.append(acc)
assert cdf[0] == pv[0]
assert cdf[1] == pv[0] + pv[1]
assert all(cdf[k] <= cdf[k + 1] for k in range(99))
assert cdf[-1] == 1
```

CLAIM c3: X defined by (X - 0.01)/(0.02 - 0.01) = U with U ~ Uniform(0,1) is Uniform(0.01, 0.02).
```python
from sympy import Rational, Symbol, solve, simplify, integrate, Piecewise, diff
from sympy.stats import Uniform, density, cdf as scdf

a, b = Rational(1, 100), Rational(2, 100)
X = Symbol('X', real=True)
u = Symbol('u', real=True)

# solve the booklet's equation for X
sol = solve((X - a) / (b - a) - u, X)
assert len(sol) == 1
Xu = sol[0]
assert simplify(Xu - (a + (b - a) * u)) == 0

# endpoints: u=0 -> 0.01, u=1 -> 0.02, and the map is increasing
assert Xu.subs(u, 0) == a
assert Xu.subs(u, 1) == b
assert simplify(diff(Xu, u) - (b - a)) == 0 and (b - a) > 0

# CDF of X: P(X <= x) = P(u <= (x-a)/(b-a)) = (x-a)/(b-a) on [a,b]
x = Symbol('x', real=True)
U = Uniform('U', 0, 1)
F = scdf(U)((x - a) / (b - a))
Fmid = simplify(F.subs(x, Rational(3, 200)))  # x = 0.015, the bin midpoint
assert Fmid == Rational(1, 2)

# density is constant 1/(b-a) = 100 on (a, b) and integrates to 1
f = simplify(diff((x - a) / (b - a), x))
assert f == Rational(100)
assert integrate(f, (x, a, b)) == 1
```
