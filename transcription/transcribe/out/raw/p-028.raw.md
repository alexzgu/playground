CLAIM c1: Example (1) — score of the normal in $\mu$ is $(x-\mu)/\sigma^2$, and $\frac{1}{\sigma^2}E(x-\mu)^2 = 1$.
```python
from sympy import *

x, mu = symbols('x mu', real=True)
sigma = symbols('sigma', positive=True)

L = (1/(2*pi*sigma**2))**Rational(1, 2) * exp(-(x - mu)**2/(2*sigma**2))
logL = log(L)

# The booklet drops the additive constant: logL(mu|x) = -(x-mu)^2/(2 sigma^2) + const
assert simplify(diff(logL, mu) - (x - mu)/sigma**2) == 0

# E(x-mu)^2 = sigma^2 under X ~ N(mu, sigma^2), so (1/sigma^2) E(x-mu)^2 = 1
E_sq = integrate((x - mu)**2 * (1/(sqrt(2*pi)*sigma)) * exp(-(x - mu)**2/(2*sigma**2)),
                 (x, -oo, oo))
assert simplify(E_sq - sigma**2) == 0
assert simplify(E_sq/sigma**2 - 1) == 0
```

CLAIM c2: Example (2) — Jeffreys prior for $\sigma^2$ with $\mu$ known is $\propto 1/\sigma^2$.
```python
from sympy import *

v = symbols('v', positive=True)          # v = sigma^2
s = symbols('s', positive=True)          # s stands for (x - mu)^2, constant in v

logL = -Rational(1, 2)*log(v) - s/(2*v)

d2 = diff(logL, v, 2)
# Fisher information: I(v) = -E[d2], with E[(x-mu)^2] = E[s] = v
I = simplify(-d2.subs(s, v))
assert simplify(I - 1/(2*v**2)) == 0

# Jeffreys prior: pi(v) ∝ sqrt(I(v)) = 1/(sqrt(2) v) ∝ 1/v = 1/sigma^2
assert simplify(sqrt(I) * sqrt(2) * v - 1) == 0
```

CLAIM c3: Example (3) — the joint Jeffreys prior for $(\mu,\sigma^2)$ is $\propto (1/\sigma^2)^{3/2}$.
```python
from sympy import *

x, mu, t = symbols('x mu t', real=True)
v = symbols('v', positive=True)          # v = sigma^2

logL = -Rational(1, 2)*log(2*pi*v) - (x - mu)**2/(2*v)

def expect(expr):
    # E over x ~ N(mu, v): integrate against the density with x = mu + t
    dens = exp(-t**2/(2*v))/sqrt(2*pi*v)
    return simplify(integrate(expr.subs(x, mu + t)*dens, (t, -oo, oo)))

I = Matrix(2, 2, lambda i, j: -expect(diff(logL, [mu, v][i], [mu, v][j])))
assert simplify(I - Matrix([[1/v, 0], [0, 1/(2*v**2)]])) == zeros(2, 2)

det_I = simplify(I.det())
assert simplify(det_I - 1/(2*v**3)) == 0

# |I|^{1/2} = (1/(2 v^3))^{1/2} = (1/sqrt(2)) * (1/v)^{3/2}
assert simplify(sqrt(det_I) * sqrt(2) - (1/v)**Rational(3, 2)) == 0
```
