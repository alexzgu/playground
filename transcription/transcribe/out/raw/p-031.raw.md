phi = symbols('phi', real=True)
p = exp(phi)/(1 + exp(phi))

# the booklet's intermediate expression
mid = (exp(phi)*(1 + exp(phi)) - exp(2*phi)) / (1 + exp(phi))**2
# the booklet's final expression
fin = exp(phi) / (1 + exp(phi))**2

assert simplify(diff(p, phi) - mid) == 0

CLAIM c2: The product $\frac{1}{\pi\sqrt{e^{\phi}/(1+e^{\phi})^{2}}}\cdot\frac{e^{\phi}}{(1+e^{\phi})^{2}}$ equals $\frac{\sqrt{e^{\phi}}}{\pi(1+e^{\phi})}$; the printed right-hand side $\frac{\sqrt{e^{\phi}}}{1+e^{\phi}}$ omits the factor $1/\pi$.

CLAIM c4: For $X\sim N(\mu,\sigma^{2})$ with $\mu$ known and $v=\sigma^{2}$, $I(v)=1/(2v^{2})$, so $\sqrt{I(v)}\propto 1/\sigma^{2}$.
```python
from sympy import *

x, mu, t = symbols('x mu t', real=True)
v = symbols('v', positive=True)          # v = sigma^2

logf = -log(2*pi*v)/2 - (x - mu)**2/(2*v)
d2 = diff(logf, v, 2)                    # 1/(2v^2) - (x-mu)^2/v^3

# take expectations via t = x - mu with E[t^2] = v
# (sympy canonicalizes (x-mu)**2 as (mu-x)**2, so a direct subs misses)
d2t = expand(d2.subs(x, mu + t))
I_v = simplify(-d2t.subs(t**2, v).subs(t, 0))
assert simplify(I_v - 1/(2*v**2)) == 0

# sqrt(I_v) = 1/(sqrt(2) v)  ->  proportional to 1/sigma^2 with constant ratio
ratio = simplify(sqrt(I_v) * v)
assert simplify(ratio - 1/sqrt(2)) == 0
```

CLAIM c5: For the binomial log-likelihood in the logit parameter, $\ell(\phi)=\phi y-n\log(1+e^{\phi})$, the Fisher information is $I(\phi)=ne^{\phi}/(1+e^{\phi})^{2}$ and $\sqrt{I(\phi)}\propto\sqrt{e^{\phi}}/(1+e^{\phi})$.
```python
from sympy import *

phi = symbols('phi', real=True)
n, y = symbols('n y', positive=True)

p = exp(phi)/(1 + exp(phi))
# log of  C(n,y) p^y (1-p)^(n-y)  , dropping the constant binomial coefficient
loglik = y*log(p) + (n - y)*log(1 - p)
assert simplify(loglik - (phi*y - n*log(1 + exp(phi)))) == 0

d2 = diff(phi*y - n*log(1 + exp(phi)), phi, 2)   # free of y already
I_phi = -d2
# avoid simplify() here (it rewrites to cosh and cannot close);
# rational-function cancel decides equality of these exp-rational expressions
assert cancel(together(I_phi - n*exp(phi)/(1 + exp(phi))**2)) == 0

# the margin's intermediate step
step = -(n*exp(phi)*(1 + exp(phi)) - n*exp(phi)*exp(phi))/(1 + exp(phi))**2
assert cancel(together(step - d2)) == 0

# sqrt(I) is a constant multiple of sqrt(e^phi)/(1+e^phi): the squared ratio is n
ratio_sq = cancel(I_phi * (1 + exp(phi))**2 / exp(phi))
assert simplify(ratio_sq - n) == 0
```

CLAIM c6: For $X\sim N(\mu,\sigma^{2})$ with both parameters unknown, $\sqrt{|I(\mu,\sigma^{2})|}\propto(\sigma^{2})^{-3/2}$.
```python
from sympy import *

x, mu, t = symbols('x mu t', real=True)
v = symbols('v', positive=True)          # v = sigma^2

logf = -log(2*pi*v)/2 - (x - mu)**2/(2*v)

def expect(e):
    # E[x-mu] = 0, E[(x-mu)^2] = v ; substitute via t = x - mu
    # (sympy canonicalizes (x-mu)**2 as (mu-x)**2, so a direct subs misses)
    e = expand(e.subs(x, mu + t))
    return e.subs(t**2, v).subs(t, 0)

I = Matrix(2, 2, lambda i, j: expect(-diff(logf, [mu, v][i], [mu, v][j])))
I = simplify(I)

assert simplify(I - Matrix([[1/v, 0], [0, 1/(2*v**2)]])) == zeros(2, 2)

detI = simplify(I.det())
assert simplify(detI - 1/(2*v**3)) == 0

# sqrt(|I|) = 1/(sqrt(2) v^{3/2})  ->  proportional to (sigma^2)^{-3/2}
ratio = simplify(sqrt(detI) * v**Rational(3, 2))
assert simplify(ratio - 1/sqrt(2)) == 0
```
