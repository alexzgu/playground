CLAIM c2: Under conditional independence, $f(y_{(r)}\mid\theta)\,[f(y\mid\theta)]^{-1} = [f(y_r\mid\theta)]^{-1}$, so the weight reduces to the Remark's form.
```python
from sympy import *

y1, y2, y3, mu = symbols('y1 y2 y3 mu', real=True)
sig = symbols('sigma', positive=True)

def f(y):
    return exp(-(y - mu)**2 / (2*sig**2)) / sqrt(2*pi*sig**2)

# n = 3, drop r = 2
f_full = f(y1) * f(y2) * f(y3)          # f(y | theta)
f_drop = f(y1) * f(y3)                   # f(y_(2) | theta)

assert simplify(f_drop / f_full - 1 / f(y2)) == 0

# hence the full weight equals the reduced weight, for M = 2 draws
mu1, mu2 = symbols('mu1 mu2', real=True)
s1, s2 = symbols('s1 s2', positive=True)

def g(y, m, s):
    return exp(-(y - m)**2 / (2*s**2)) / sqrt(2*pi*s**2)

full = [(g(y1, m, s) * g(y3, m, s)) / (g(y1, m, s) * g(y2, m, s) * g(y3, m, s))
        for m, s in [(mu1, s1), (mu2, s2)]]
red = [1 / g(y2, m, s) for m, s in [(mu1, s1), (mu2, s2)]]

w_full = [t / sum(full) for t in full]
w_red = [t / sum(red) for t in red]
for a_, b_ in zip(w_full, w_red):
    assert simplify(a_ - b_) == 0
```

CLAIM c3: With $f(y_{ij}\mid\Omega^{(h)})$ the $N(\mu_i^{(h)},\sigma^{2(h)})$ density, the weights equal the displayed explicit expression and sum to 1.
```python
from sympy import *
from sympy.stats import Normal, density

y = Symbol('y', real=True)
mus = symbols('m1 m2 m3', real=True)
# sigma^{2(h)} are the variances; write them as squares of positive symbols
sds = symbols('s1 s2 s3', positive=True)

# the booklet's printed density, with sigma^{2(h)} = s_h**2
def booklet_density(yv, m, s):
    return (1 / sqrt(2 * pi * s**2)) * exp(-(yv - m)**2 / (2 * s**2))

# check the printed form is exactly the N(m, s^2) density
for m, s in zip(mus, sds):
    X = Normal('X', m, s)
    assert simplify(density(X)(y) - booklet_density(y, m, s)) == 0

# weights: w_h = (1/f_h) / sum_h (1/f_h)
inv = [1 / booklet_density(y, m, s) for m, s in zip(mus, sds)]
w = [t / sum(inv) for t in inv]

assert simplify(sum(w) - 1) == 0

# and equals the fully-written-out printed ratio
printed = [
    (1 / ((1 / sqrt(2 * pi * s**2)) * exp(-(y - m)**2 / (2 * s**2))))
    / sum(1 / ((1 / sqrt(2 * pi * sk**2)) * exp(-(y - mk)**2 / (2 * sk**2)))
          for mk, sk in zip(mus, sds))
    for m, s in zip(mus, sds)
]
for a_, b_ in zip(w, printed):
    assert simplify(a_ - b_) == 0
```
