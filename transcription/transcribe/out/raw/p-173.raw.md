CLAIM c1: The printed SEP density at $\lambda=0$, $\alpha=1$ equals the $N(\mu,\sigma^2)$ pdf.
```python
from sympy import *

y, mu = symbols('y mu', real=True)
sigma = symbols('sigma', positive=True)

def Phi(x):
    return (1 + erf(x/sqrt(2)))/2

# printed density, evaluated at lambda = 0, alpha = 1
alpha = Integer(1)
lam = Integer(0)
z = (y - mu)/sigma

const = 1/(2*sigma*alpha**(Rational(1,1)/alpha - 1)*gamma(1/alpha))
kernel = exp(-Abs(z)**(2*alpha)/(2*alpha))
# sign(lam*z)*|lam*z|^alpha = lam*z when alpha = 1; with lam = 0 the argument is 0
skew = Phi(Integer(0))
printed = const*kernel*skew

normal = 1/(sigma*sqrt(2*pi))*exp(-(y - mu)**2/(2*sigma**2))

# the booklet asserts printed == N(mu, sigma^2)
assert simplify(printed - normal) == 0
```
