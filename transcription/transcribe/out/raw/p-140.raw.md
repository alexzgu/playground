CLAIM c1: Marginalizing the slice-sampler augmentation over $\mu_1,\mu_2$ returns the target kernel $e^{-x^2/2}e^x/(1+e^x)^2$.
```python
from sympy import *

x, m1, m2 = symbols('x mu1 mu2', real=True)

a = exp(-x**2/2)                 # first slice height
b = exp(x)/(1 + exp(x))**2       # second slice height

# Both heights are strictly positive for every real x, so the slice regions are non-degenerate.
assert simplify(a) == exp(-x**2/2)
for val in [-3, -1, 0, Rational(1, 2), 2]:
    assert a.subs(x, val) > 0 and b.subs(x, val) > 0

# Integrate the indicator joint over mu1 in (0, a) and mu2 in (0, b).
inner = integrate(1, (m2, 0, b))
marginal = integrate(inner, (m1, 0, a))

target = exp(-x**2/2) * exp(x)/(1 + exp(x))**2
assert simplify(marginal - target) == 0
```

CLAIM c2: The full conditionals of the augmented joint are $U(0,e^{-x^2/2})$ and $U(0,e^x/(1+e^x)^2)$.

from sympy import *

x, m1, m2 = symbols('x mu1 mu2', real=True)

a = exp(-x**2/2)
b = exp(x)/(1 + exp(x))**2

# Conditional on x, mu2, the joint is constant in mu1 on (0, a):
# normalizing 1 over that interval gives density 1/a there.
Z1 = integrate(1, (m1, 0, a))
assert simplify(Z1 - a) == 0
dens1 = 1/Z1
assert simplify(dens1 - exp(x**2/2)) == 0          # = 1/a, the U(0, a) density

# Symmetrically for mu2 on (0, b).

CLAIM c3: $\pi(u,\theta,\phi\mid y)=\pi(u\mid\theta,\phi,y)\pi(\theta\mid\phi,y)\pi(\phi\mid y)$ holds exactly (chain rule) on an arbitrary discrete joint.
```python
from sympy import Rational, simplify
from itertools import product

# A fully generic 2x2x2 joint over (u, theta, phi) given fixed y, with exact rationals.
w = [Rational(k) for k in [3, 1, 4, 1, 5, 9, 2, 6]]
tot = sum(w)
p = {s: w[i]/tot for i, s in enumerate(product([0, 1], repeat=3))}
assert simplify(sum(p.values()) - 1) == 0

def marg(keep):  # marginal over the retained coordinate indices
    out = {}
    for s, v in p.items():
        k = tuple(s[i] for i in keep)
        out[k] = out.get(k, 0) + v
    return out

p_tp = marg([1, 2])   # pi(theta, phi | y)
p_p = marg([2])       # pi(phi | y)

for u, t, f in p:
    lhs = p[(u, t, f)]
    rhs = (p[(u, t, f)]/p_tp[(t, f)]) * (p_tp[(t, f)]/p_p[(f,)]) * p_p[(f,)]
    assert simplify(lhs - rhs) == 0
```
