CLAIM c1: Under the prior $\pi(\mu,\sigma^2)\propto 1/\sigma^2$, integrating $\mu$ out of the normal posterior leaves a kernel in $\sigma^2$ whose change of variables $t=(n-1)s^2/\sigma^2$ is exactly the $\chi^2_{n-1}$ density.
```python
from sympy import *

mu, sigma2, ybar, t = symbols('mu sigma2 ybar t', real=True)
sigma2 = Symbol('sigma2', positive=True)
t = Symbol('t', positive=True)

n = 5                 # exact small values
s2 = Rational(3)      # s^2
nu = n - 1            # degrees of freedom

# joint posterior kernel: prior 1/sigma^2 times N(mu, sigma^2) likelihood,
# with sum (y_i - mu)^2 = (n-1)s^2 + n (mu - ybar)^2
post = sigma2**(-1) * sigma2**(-Rational(n, 2)) * \
       exp(-((nu * s2 + n * (mu - ybar)**2) / (2 * sigma2)))

marg = integrate(post, (mu, -oo, oo))
marg = simplify(marg)

# claimed marginal kernel for sigma^2
claimed = sigma2**(-Rational(n + 1, 2)) * exp(-nu * s2 / (2 * sigma2))

# the two must agree up to a constant free of sigma2 and ybar
ratio = simplify(marg / claimed)
assert simplify(diff(ratio, sigma2)) == 0
assert simplify(diff(ratio, ybar)) == 0

# change of variables t = (n-1)s^2/sigma^2  =>  sigma^2 = nu*s2/t
s2_of_t = nu * s2 / t
jac = Abs(diff(s2_of_t, t))
g = simplify(claimed.subs(sigma2, s2_of_t) * jac)

# normalize and compare to the chi-square(nu) density
Z = integrate(g, (t, 0, oo))
g_norm = simplify(g / Z)

chi2_pdf = t**(Rational(nu, 2) - 1) * exp(-t / 2) / (2**Rational(nu, 2) * gamma(Rational(nu, 2)))

assert simplify(g_norm - chi2_pdf) == 0
```

CLAIM c2: The conditional posterior of $\mu$ given $\sigma^2$ and the data is exactly $N(\bar y,\sigma^2/n)$.
```python
from sympy import *

mu = Symbol('mu', real=True)
ybar = Symbol('ybar', real=True)
sigma2 = Symbol('sigma2', positive=True)
n = Symbol('n', positive=True, integer=True)

nval = 5
s2 = Rational(3)

# joint posterior kernel viewed as a function of mu (sigma^2 held fixed)
kernel = exp(-((nval - 1) * s2 + nval * (mu - ybar)**2) / (2 * sigma2))

Z = integrate(kernel, (mu, -oo, oo))
cond = simplify(kernel / Z)

# claimed density: N(ybar, sigma^2 / n)
var = sigma2 / nval
normal_pdf = exp(-(mu - ybar)**2 / (2 * var)) / sqrt(2 * pi * var)

assert simplify(cond - normal_pdf) == 0
```

CLAIM c3: $\lambda=\phi/\sqrt{1-\phi^2}$ is a strictly increasing bijection from $(-1,1)$ onto $\mathbb{R}$ with inverse $\phi=\lambda/\sqrt{1+\lambda^2}$.
```python
from sympy import *

phi = Symbol('phi', real=True)
lam = Symbol('lam', real=True)

lam_of_phi = phi / sqrt(1 - phi**2)
phi_of_lam = lam / sqrt(1 + lam**2)

# round-trip identity, checked exactly at interior rational points of (-1, 1)
for val in [Rational(-9, 10), Rational(-1, 3), Integer(0), Rational(1, 3), Rational(1, 2), Rational(9, 10)]:
    l = lam_of_phi.subs(phi, val)
    assert simplify(phi_of_lam.subs(lam, l) - val) == 0

# and the reverse round trip
for val in [Integer(-4), Rational(-1, 2), Integer(0), Rational(7, 3), Integer(5)]:
    p = phi_of_lam.subs(lam, val)
    assert simplify(lam_of_phi.subs(phi, p) - val) == 0

# strictly increasing on (-1, 1): derivative (1 - phi^2)^(-3/2) > 0
deriv = simplify(diff(lam_of_phi, phi))
assert simplify(deriv - (1 - phi**2)**Rational(-3, 2)) == 0
for val in [Rational(-9, 10), Integer(0), Rational(9, 10)]:
    assert deriv.subs(phi, val) > 0

# range is all of R: endpoints blow up with the right signs
assert limit(lam_of_phi, phi, 1, '-') == oo
assert limit(lam_of_phi, phi, -1, '+') == -oo
```

CLAIM c4: $\gamma=(\phi+1)/2$ carries $(-1,1)$ bijectively onto $(0,1)$, and $\rho=1/\alpha$ is equivalent to $\alpha=1/\rho$ (with $0<\rho<1 \iff \alpha>1$).
```python
from sympy import *

phi = Symbol('phi', real=True)
alpha = Symbol('alpha', positive=True)
rho = Symbol('rho', positive=True)

gam = (phi + 1) / 2

# affine, strictly increasing, endpoints -1 -> 0 and 1 -> 1
assert simplify(diff(gam, phi) - Rational(1, 2)) == 0
assert simplify(gam.subs(phi, -1) - 0) == 0
assert simplify(gam.subs(phi, 1) - 1) == 0

# inverse map phi = 2*gamma - 1 is a genuine two-sided inverse
g = Symbol('g', real=True)
assert simplify(gam.subs(phi, 2 * g - 1) - g) == 0
assert simplify((2 * gam - 1) - phi) == 0

# interior points land strictly inside (0, 1)
for val in [Rational(-9, 10), Rational(-1, 3), Integer(0), Rational(1, 2), Rational(9, 10)]:
    gv = gam.subs(phi, val)
    assert 0 < gv < 1

# rho = 1/alpha and alpha = 1/rho are the same relation (an involution)
assert simplify(1 / (1 / alpha) - alpha) == 0
assert simplify(1 / (1 / rho) - rho) == 0

# 0 < rho < 1  <=>  alpha > 1
for a in [Rational(3, 2), Integer(2), Integer(10)]:
    assert 0 < Rational(1, 1) / a < 1
for a in [Rational(1, 2), Rational(9, 10)]:
    assert not (Rational(1, 1) / a < 1)
```
