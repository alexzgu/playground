CLAIM c1: The printed kernel $\left[1+\frac{((\mu-\mu_n)/(\sigma_n/\sqrt{k_n}))^2}{\nu_n}\right]^{-(\nu_n+1)/2}$ is proportional to the Student-$t_{\nu_n}$ density with location $\mu_n$ and scale $\sigma_n/\sqrt{k_n}$.
```python
from sympy import *

mu, mun, sn, kn = symbols('mu mu_n sigma_n k_n', real=True)
sn, kn = symbols('sigma_n k_n', positive=True)
mu, mun = symbols('mu mu_n', real=True)

# exact rational degrees of freedom so the integral closes
for nu in [Integer(1), Integer(2), Integer(5), Integer(8)]:
    s = sn/sqrt(kn)                       # scale
    kernel = (1 + ((mu - mun)/s)**2/nu)**(-(nu+1)/2)
    # official t density with location mun, scale s, df nu
    dens = gamma((nu+1)/2)/(gamma(nu/2)*sqrt(nu*pi)*s) * kernel
    # (a) dens integrates to 1  => kernel is proportional to the t density
    total = integrate(dens.subs({mun: 0, sn: 1, kn: 1}), (mu, -oo, oo))
    assert simplify(total - 1) == 0, (nu, total)
    # (b) kernel / dens is free of mu (i.e. they are proportional)
    ratio = simplify(kernel/dens)
    assert diff(ratio, mu) == 0
```

CLAIM c2: $w=\sigma_2^2/(\sigma_1^2+\sigma_2^2)$ uniquely minimizes $w^2\sigma_1^2+(1-w)^2\sigma_2^2$ and lies in $[0,1]$.
```python
from sympy import *

w = symbols('w', real=True)
s1, s2 = symbols('sigma_1 sigma_2', positive=True)

g = w**2*s1**2 + (1-w)**2*s2**2
crit = solve(Eq(diff(g, w), 0), w)
assert len(crit) == 1
wstar = simplify(crit[0])

assert simplify(wstar - s2**2/(s1**2 + s2**2)) == 0
assert diff(g, w, 2).is_positive        # strictly convex -> unique global min
# interior of [0,1]
assert simplify(wstar) .subs({s1: Rational(3,2), s2: Rational(7,5)}) > 0
assert (wstar < 1) == True or simplify(1 - wstar) == s1**2/(s1**2+s2**2)
assert simplify(1 - wstar - s1**2/(s1**2 + s2**2)) == 0
```

CLAIM c3: Completing the square: the residual after extracting $\left(\frac{1}{\sigma_1^2}+\frac{1}{\sigma_2^2}\right)(\theta-m)^2$ is free of $\theta$.
```python
from sympy import *

th, t1, t2 = symbols('theta thetahat_1 thetahat_2', real=True)
s1, s2 = symbols('sigma_1 sigma_2', positive=True)

p1, p2 = 1/s1**2, 1/s2**2
lhs = p1*(th - t1)**2 + p2*(th - t2)**2
m = (p1*t1 + p2*t2)/(p1 + p2)
rhs = (p1 + p2)*(th - m)**2

residual = simplify(expand(lhs - rhs))
assert simplify(diff(residual, th)) == 0        # no theta dependence
# and the residual is the expected constant
expected = p1*p2/(p1 + p2)*(t1 - t2)**2
assert simplify(residual - expected) == 0
```

CLAIM c4: Bayesian posterior mean/variance coincide with the optimally-weighted non-Bayesian combination.
```python
from sympy import *

t1, t2 = symbols('thetahat_1 thetahat_2', real=True)
s1, s2 = symbols('sigma_1 sigma_2', positive=True)

p1, p2 = 1/s1**2, 1/s2**2
post_mean = (p1*t1 + p2*t2)/(p1 + p2)
post_var  = 1/(p1 + p2)

w = s2**2/(s1**2 + s2**2)
freq_est = w*t1 + (1 - w)*t2
freq_var = w**2*s1**2 + (1 - w)**2*s2**2

assert simplify(post_mean - freq_est) == 0
assert simplify(post_var - freq_var) == 0
# and this common variance is the harmonic-type combination
assert simplify(post_var - s1**2*s2**2/(s1**2 + s2**2)) == 0
```
