CLAIM c3: $0 < e^{y\nu}/(1+e^{\nu})^{m} < 1$ for all real $\nu$ and integers $0\le y\le m$, $m\ge 1$.
```python
from sympy import *

t = symbols('t', positive=True)   # t = exp(nu) ranges over (0, oo)

for m in range(1, 8):
    for y in range(0, m + 1):
        # lower bound: numerator e^{y*nu} = t**y > 0, denominator > 0
        assert (t ** y).is_positive is True
        assert ((1 + t) ** m).is_positive is True

        # upper bound: (1+t)^m - t^y > 0 for all t > 0
        p = Poly(expand((1 + t) ** m - t ** y), t)
        coeffs = p.all_coeffs()
        # every coefficient nonnegative ...
        assert all(c >= 0 for c in coeffs)
        # ... and at least one strictly positive coefficient at a degree != y,
        # so the polynomial is strictly positive on t > 0
        strict = [d for (d,), c in p.terms() if c > 0 and d != y]
        assert len(strict) >= 1

# spot-check exact values stay strictly inside (0, 1)
for m in range(1, 6):
    for y in range(0, m + 1):
        for tv in [Rational(1, 10), Rational(1, 1), Rational(37, 5), Rational(100, 1)]:
            val = tv ** y / (1 + tv) ** m
            assert val > 0 and val < 1
```
