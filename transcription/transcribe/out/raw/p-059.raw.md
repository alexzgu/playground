CLAIM c1: The Bernoulli likelihood with logistic success probability equals $\prod_i e^{y_i x_i'\beta}/(1+e^{x_i'\beta})$.
```python
from sympy import *

u = symbols('u', real=True)
p = exp(u) / (1 + exp(u))

# y is 0/1 valued; check both cases exactly
for y in (0, 1):
    lhs = p**y * (1 - p)**(1 - y)
    rhs = exp(y * u) / (1 + exp(u))
    assert simplify(lhs - rhs) == 0

# hence the n-fold product agrees term by term
assert simplify((1 - p) - 1/(1 + exp(u))) == 0
```

CLAIM c2: $\ell(\beta) = (\sum_i y_i x_i')\beta - \sum_i \log(1+e^{x_i'\beta})$ is the log of $\prod_i e^{y_i x_i'\beta}/(1+e^{x_i'\beta})$.
```python
from sympy import *

n, p = 3, 2
b = Matrix(symbols('b1 b2', real=True))
xs = [Matrix(symbols(f'x{i}1 x{i}2', real=True)) for i in range(1, n + 1)]
ys = symbols('y1 y2 y3', real=True)

us = [(x.T * b)[0] for x in xs]

prod_expr = 1
for y, u in zip(ys, us):
    prod_expr *= exp(y * u) / (1 + exp(u))

ell = sum(y * u for y, u in zip(ys, us)) - sum(log(1 + exp(u)) for u in us)

# exp(ell) == product  <=>  ell == log(product), avoiding log-branch issues
assert simplify(exp(ell) - prod_expr) == 0
```

CLAIM c3: $\nabla_\beta \ell(\beta) = \sum_i y_i x_i - \sum_i x_i e^{x_i'\beta}/(1+e^{x_i'\beta})$.
```python
from sympy import *

n = 3
b1, b2 = symbols('b1 b2', real=True)
b = Matrix([b1, b2])
xs = [Matrix(symbols(f'x{i}1 x{i}2', real=True)) for i in range(1, n + 1)]
ys = symbols('y1 y2 y3', real=True)
us = [(x.T * b)[0] for x in xs]

ell = sum(y * u for y, u in zip(ys, us)) - sum(log(1 + exp(u)) for u in us)

grad = Matrix([diff(ell, v) for v in (b1, b2)])

rhs = zeros(2, 1)
```

CLAIM c4: $\nabla^2_\beta \ell(\beta) = -\sum_i x_i x_i' e^{x_i'\beta}/(1+e^{x_i'\beta})^2 = H$.
```python
from sympy import *

n = 3
b1, b2 = symbols('b1 b2', real=True)
b = Matrix([b1, b2])
xs = [Matrix(symbols(f'x{i}1 x{i}2', real=True)) for i in range(1, n + 1)]
ys = symbols('y1 y2 y3', real=True)
us = [(x.T * b)[0] for x in xs]

ell = sum(y * u for y, u in zip(ys, us)) - sum(log(1 + exp(u)) for u in us)

H_sym = hessian(ell, (b1, b2))

H_rhs = zeros(2, 2)
for x, u in zip(xs, us):
    H_rhs += -(x * x.T) * exp(u) / (1 + exp(u))**2

assert simplify(H_sym - H_rhs) == zeros(2, 2)
```

CLAIM c5: $\hat\beta = (X'X)^{-1}X'y$ satisfies the normal equations $X'(y - X\hat\beta) = 0$.
```python
from sympy import *

X = Matrix([[1, 2], [1, 5], [1, 7]])   # exact rationals, full column rank
y = Matrix(symbols('y1 y2 y3', real=True))

bhat = (X.T * X).inv() * X.T * y
assert simplify(X.T * (y - X * bhat)) == zeros(2, 1)

# and it is the least-squares minimizer: gradient of ||y - Xb||^2 vanishes at bhat
b1, b2 = symbols('b1 b2', real=True)
b = Matrix([b1, b2])
ssq = ((y - X * b).T * (y - X * b))[0]
grad = Matrix([diff(ssq, v) for v in (b1, b2)])
assert simplify(grad.subs({b1: bhat[0], b2: bhat[1]})) == zeros(2, 1)
```
