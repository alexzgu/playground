CLAIM c1: Dirichlet(1,...,1) has a density that is constant on the simplex (uniform prior), equal to (k-1)!.
```python
from sympy import *

# k = 4 (matches the two-way table case theta = (t11,t12,t21,t22))
k = 4
t1, t2, t3 = symbols('t1 t2 t3', positive=True)
t4 = 1 - t1 - t2 - t3

alpha = [Integer(1)] * k
tau = sum(alpha)

# Dirichlet density on the simplex, expressed in the free coordinates
B = prod([gamma(a) for a in alpha]) / gamma(tau)
dens = simplify(prod([t**(a - 1) for t, a in zip([t1, t2, t3, t4], alpha)]) / B)

# constant in every free coordinate  -> uniform
assert simplify(diff(dens, t1)) == 0
assert simplify(diff(dens, t2)) == 0
assert simplify(diff(dens, t3)) == 0
assert simplify(dens - factorial(k - 1)) == 0   # (k-1)! = 6

# and it is a proper density: integrates to 1 over the simplex
I = integrate(integrate(integrate(dens, (t3, 0, 1 - t1 - t2)), (t2, 0, 1 - t1)), (t1, 0, 1))
assert simplify(I - 1) == 0
```

CLAIM c2: Multinomial likelihood times Dirichlet(alpha) prior yields exactly the Dirichlet(n+alpha) posterior.
```python
from sympy import *

# k = 3, exact integer counts and hyperparameters so the simplex integrals
# are polynomial and close quickly (the fractional-alpha version times out)
t1, t2 = symbols('t1 t2', positive=True)
t3 = 1 - t1 - t2
th = [t1, t2, t3]

nvec = [Integer(2), Integer(3), Integer(4)]
n = sum(nvec)
alpha = [Integer(1), Integer(2), Integer(3)]

def dirichlet_density(a):
    B = prod([gamma(x) for x in a]) / gamma(sum(a))
    return prod([t**(x - 1) for t, x in zip(th, a)]) / B

lik = factorial(n) / prod([factorial(x) for x in nvec]) * prod([t**x for t, x in zip(th, nvec)])
prior = dirichlet_density(alpha)

joint = expand(lik * prior)
marginal = integrate(integrate(joint, (t2, 0, 1 - t1)), (t1, 0, 1))
posterior = simplify(joint / marginal)

claimed = dirichlet_density([x + y for x, y in zip(nvec, alpha)])   # Dirichlet(n + alpha)
assert simplify(posterior - claimed) == 0

# and the claimed posterior is a proper density on the simplex
assert simplify(integrate(integrate(expand(claimed), (t2, 0, 1 - t1)), (t1, 0, 1)) - 1) == 0

# kernel-level conjugacy for GENERAL alpha: prod t^n_i * prod t^(a_i-1) is the
# Dirichlet(n+alpha) kernel identically (no integration needed)
a1, a2, a3 = symbols('a1 a2 a3', positive=True)
gen_joint = prod([t**x for t, x in zip(th, nvec)]) * prod([t**(a - 1) for t, a in zip(th, [a1, a2, a3])])
gen_claimed = prod([t**(x + a - 1) for t, x, a in zip(th, nvec, [a1, a2, a3])])
assert simplify(gen_joint - gen_claimed) == 0
```

CLAIM c3: Jeffreys prior for the multinomial is proportional to prod theta_i^{-1/2}, i.e. Dirichlet(alpha_i = 1/2).
```python
from sympy import *

n = symbols('n', positive=True)
# use a positive symbol for t3 too, so sqrt() factors cleanly;
# the simplex constraint t1 + t2 + t3 = 1 is imposed by substitution below
t1, t2, t3 = symbols('t1 t2 t3', positive=True)
th = [t1, t2, t3]

# Fisher information for Multi(n, theta) in the free coords (t1, t2), t3 = 1 - t1 - t2:
# I_ij = n * (delta_ij / t_i + 1 / t_3)
I = n * Matrix([[1/t1 + 1/t3, 1/t3],
                [1/t3,        1/t2 + 1/t3]])

detI = simplify(I.det())
# det I = n^2 (t1 + t2 + t3)/(t1 t2 t3), which is n^2/(t1 t2 t3) on the simplex
assert simplify(detI - n**2*(t1 + t2 + t3)/(t1*t2*t3)) == 0
assert simplify(detI.subs(t3, 1 - t1 - t2) - n**2/(t1*t2*(1 - t1 - t2))) == 0

jeffreys = sqrt(detI)                      # ∝ (t1 t2 t3)^(-1/2) on the simplex
dir_half_kernel = prod([t**(Rational(1, 2) - 1) for t in th])

# proportional: the ratio is free of theta once the simplex constraint is used
ratio = simplify(jeffreys / dir_half_kernel)
assert simplify(ratio - n*sqrt(t1 + t2 + t3)) == 0
on_simplex = simplify(ratio.subs(t3, 1 - t1 - t2))
assert simplify(diff(on_simplex, t1)) == 0
assert simplify(diff(on_simplex, t2)) == 0
assert simplify(on_simplex - n) == 0
```
