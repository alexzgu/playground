CLAIM c1: For p ~ Beta(μτ, (1-μ)τ), E(p) = μ and Var(p) = μ(1-μ)/(τ+1).
```python
from sympy import symbols, simplify, Rational, integrate, gamma

mu, tau, p = symbols('mu tau p', positive=True)
a = mu*tau
b = (1-mu)*tau

# Beta(a,b) mean and variance in terms of a,b
mean_ab = a/(a+b)
var_ab = a*b/((a+b)**2*(a+b+1))

assert simplify(mean_ab - mu) == 0
assert simplify(var_ab - mu*(1-mu)/(tau+1)) == 0

# cross-check by direct integration with exact Rational parameters
A, B = Rational(3,2), Rational(1,2)   # mu=3/4, tau=2
Bfn = gamma(A)*gamma(B)/gamma(A + B)  # Beta function via gamma, evaluates exactly
dens = p**(A-1)*(1-p)**(B-1)/Bfn
m = integrate(p*dens, (p, 0, 1))
m2 = integrate(p**2*dens, (p, 0, 1))
assert simplify(m - Rational(3,4)) == 0
assert simplify((m2 - m**2) - Rational(3,4)*Rational(1,4)/(2+1)) == 0
```
