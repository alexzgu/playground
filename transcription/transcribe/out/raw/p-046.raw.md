CLAIM c1: With prior $\pi(\mu,\sigma^2)\propto1/\sigma^2$ and iid normal data, $\sum_i(y_i-\mu)^2=(n-1)s^2+n(\bar y-\mu)^2$, and integrating $\sigma^2$ out of the joint posterior leaves a marginal for $\mu$ proportional to $(1+t^2/(n-1))^{-n/2}$ with $t=(\bar y-\mu)/(s/\sqrt n)$, i.e. the $t_{n-1}$ kernel.
```python
from sympy import *

# (a) the sum-of-squares decomposition, exactly, for n = 4
n_ = 4
y = symbols('y1:5', real=True)
mu = Symbol('mu', real=True)
ybar = Rational(1, n_) * sum(y)
s2 = Rational(1, n_ - 1) * sum((yi - ybar)**2 for yi in y)
lhs = sum((yi - mu)**2 for yi in y)
rhs = (n_ - 1) * s2 + n_ * (ybar - mu)**2
assert simplify(expand(lhs - rhs)) == 0

# (b) integrating sigma^2 out of  (sigma^2)^{-n/2 - 1} exp(-A / (2 sigma^2))
#     leaves a kernel proportional to A^{-n/2}.  Done exactly at n = 5.
v = Symbol('v', positive=True)      # v = sigma^2
A = Symbol('A', positive=True)      # A = sum (y_i - mu)^2
n5 = 5
marg = integrate(v**(-Rational(n5, 2) - 1) * exp(-A / (2 * v)), (v, 0, oo))
assert simplify(marg - 2**Rational(n5, 2) * gamma(Rational(n5, 2)) * A**(-Rational(n5, 2))) == 0

# (c) A^{-n/2} is the t_{n-1} kernel in t = (ybar - mu)/(s/sqrt(n))
S2, T = symbols('S2 T', positive=True)          # S2 = s^2, T = t
# A = (n-1) s^2 + n (ybar - mu)^2 = (n-1) s^2 (1 + T^2/(n-1))
A_expr = (n5 - 1) * S2 * (1 + T**2 / (n5 - 1))
kernel = A_expr**(-Rational(n5, 2))
t_kernel = (1 + T**2 / (n5 - 1))**(-Rational(n5, 2))
# proportional in T: ratio must be free of T
ratio = simplify(kernel / t_kernel)
assert simplify(diff(ratio, T)) == 0
```

CLAIM c3: The joint posterior $\pi(\beta,\sigma^2\mid y)$ is proportional to $\frac{1}{\sigma^2}(2\pi\sigma^2)^{-n/2}\exp\{-\frac{1}{2\sigma^2}\sum_{i=1}^n(y_i-x_i'\beta)^2\}$, i.e. the prior $1/\sigma^2$ times the product of $n$ independent $N(x_i'\beta,\sigma^2)$ densities.
```python
from sympy import *

# n = 3 observations, p = 2 regression coefficients: build the likelihood as an
# explicit product of normal densities and match the booklet's closed form.
n_, p_ = 3, 2
sig2 = Symbol('sigma2', positive=True)
b0, b1 = symbols('beta0 beta1', real=True)
ys = symbols('y1:4', real=True)
xs = symbols('x11 x12 x13', real=True)

def normal_pdf(yi, mean):
    return (2*pi*sig2)**Rational(-1, 2) * exp(-(yi - mean)**2 / (2*sig2))

means = [b0 + b1*xi for xi in xs]
likelihood = prod(normal_pdf(yi, m) for yi, m in zip(ys, means))

booklet = (2*pi*sig2)**Rational(-n_, 2) * exp(
    -sum((yi - m)**2 for yi, m in zip(ys, means)) / (2*sig2))
assert simplify(likelihood - booklet) == 0

# posterior = (1/sigma^2) * likelihood, exactly the printed expression
prior = 1 / sig2
posterior = prior * likelihood
printed = (1/sig2) * (1/(2*pi*sig2))**Rational(n_, 2) * exp(
    -sum((yi - m)**2 for yi, m in zip(ys, means)) / (2*sig2))
assert simplify(posterior - printed) == 0
```
