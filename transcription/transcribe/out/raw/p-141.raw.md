**12.5 Multinomial-Dirichlet model**

Perform $n$ identical independent trials.

$$n \mid \theta \sim \text{Multinomial}(n, \theta)$$

$$f(n \mid \theta) = \frac{n!}{\prod_{i=1}^{k} n_i!} \prod_{i=1}^{k} \theta_i^{n_i} \text{ where } n_i \geq 0, n = \sum_{i=1}^{k} n_i$$

from sympy import *
from itertools import product as iproduct

v = symbols('v', real=True)
p = exp(v) / (1 + exp(v))
q = 1 / (1 + exp(v))

# 1 - p must equal the booklet's second factor 1/(1+e^v)
assert simplify(1 - p - q) == 0

# collapse identity, checked exhaustively over all binary y-vectors for n_i = 1..5
for n_i in range(1, 6):
    for y in iproduct([0, 1], repeat=n_i):
        lhs = prod([p**y[j] * q**(1 - y[j]) for j in range(n_i)])
        s_i = sum(y)
        rhs = p**s_i * q**(n_i - s_i)
        assert simplify(lhs - rhs) == 0, (n_i, y)

# symbolic version with generic exponents summing to s_i
n_i, s_i = symbols('n_i s_i', positive=True)
assert simplify(p**s_i * q**(n_i - s_i) - exp(v * s_i) / (1 + exp(v))**n_i) == 0
