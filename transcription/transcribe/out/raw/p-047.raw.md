CLAIM c1: The least-squares decomposition $\sum_i (y_i-x_i'\beta)^2 = \sum_i (y_i-x_i'\hat\beta)^2 + (\beta-\hat\beta)'X'X(\beta-\hat\beta)$ holds when $\hat\beta=(X'X)^{-1}X'y$.
```python
from sympy import *

# concrete X (n=4, p=2), concrete y, symbolic beta
X = Matrix([[1, 2], [1, 5], [1, 7], [1, 11]])
y = Matrix([3, 8, 10, 17])
b1, b2 = symbols('b1 b2', real=True)
beta = Matrix([b1, b2])

bhat = (X.T * X).inv() * X.T * y

lhs = ((y - X * beta).T * (y - X * beta))[0, 0]
rhs = ((y - X * bhat).T * (y - X * bhat))[0, 0] \
    + ((beta - bhat).T * (X.T * X) * (beta - bhat))[0, 0]

assert simplify(expand(lhs - rhs)) == 0
```

CLAIM c2: $\exp\left(-\frac{1}{2\sigma^2}(\beta-\hat\beta)'X'X(\beta-\hat\beta)\right)$ is the kernel of $N(\hat\beta,\ \sigma^2(X'X)^{-1})$.
```python
from sympy import *

s2 = symbols('sigma2', positive=True)
b1, b2 = symbols('b1 b2', real=True)
beta = Matrix([b1, b2])

X = Matrix([[1, 2], [1, 5], [1, 7], [1, 11]])
XtX = X.T * X
bhat = Matrix([Rational(1, 2), Rational(3, 2)])

# covariance of the claimed normal, and its inverse
Sigma = s2 * XtX.inv()
Sinv = Sigma.inv()

# general MVN kernel exponent: -1/2 (b-bhat)' Sigma^{-1} (b-bhat)
mvn_exponent = -Rational(1, 2) * ((beta - bhat).T * Sinv * (beta - bhat))[0, 0]
# booklet's exponent
booklet_exponent = -1 / (2 * s2) * ((beta - bhat).T * XtX * (beta - bhat))[0, 0]

assert simplify(mvn_exponent - booklet_exponent) == 0

# and the claimed covariance really is sigma^2 (X'X)^{-1}
assert simplify(Sinv.inv() - s2 * XtX.inv()) == zeros(2, 2)
```

CLAIM c3: $(\sigma^2)^{-(\frac{n-p}{2}+1)}e^{-S/(2\sigma^2)}$ is the $IG\!\left(\frac{n-p}{2},\frac{S}{2}\right)$ kernel, and the corresponding density is proper.
```python
from sympy import *

x, S = symbols('x S', positive=True)
n, p = 10, 3                      # so (n-p)/2 = 7/2 > 0
a = Rational(n - p, 2)
b = S / 2

# Inverse-Gamma(a, b) density
ig = b**a / gamma(a) * x**(-(a + 1)) * exp(-b / x)

# booklet's kernel for pi(sigma^2 | y)
kernel = (1 / x)**(Rational(n - p, 2) + 1) * exp(-S / (2 * x))

# they agree up to a constant free of x
ratio = simplify(ig / kernel)
assert simplify(diff(ratio, x)) == 0

# proper density
assert simplify(integrate(ig, (x, 0, oo)) - 1) == 0
```

CLAIM c4: Marginalizing $\sigma^2$ out of the joint posterior yields the multivariate-$t$ kernel with exponent $\frac{n-p+p}{2}=\frac{n}{2}$.
```python
from sympy import *

s2 = symbols('sigma2', positive=True)
S, Q = symbols('S Q', positive=True)   # S = residual SS, Q = quadratic form in beta
n = 10

integrand = (1 / s2)**(Rational(n, 2) + 1) * exp(-(S + Q) / (2 * s2))
val = simplify(integrate(integrand, (s2, 0, oo)))

# closed form: Gamma(n/2) * 2^(n/2) * (S+Q)^(-n/2)
expected = gamma(Rational(n, 2)) * 2**Rational(n, 2) * (S + Q)**(-Rational(n, 2))
assert simplify(val - expected) == 0

# and (S+Q)^(-n/2) is proportional (in beta, i.e. in Q) to [1 + Q/S]^(-(n-p+p)/2)
p = 3
booklet = (1 + Q / S)**(-Rational((n - p) + p, 2))
ratio = simplify(expected / booklet)
assert simplify(diff(ratio, Q)) == 0        # constant in Q
```
