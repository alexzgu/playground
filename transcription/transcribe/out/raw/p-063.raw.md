CLAIM c1: The skew-normal density $f(x)=2\phi(x)\Phi(\lambda x)$ integrates to 1 over the real line.
```python
from sympy import *

x = symbols('x', real=True)

def check(lam):
    phi = exp(-x**2/2)/sqrt(2*pi)
    Phi = (1 + erf(lam*x/sqrt(2)))/2
    val = integrate(2*phi*Phi, (x, -oo, oo))
    return simplify(val - 1)

# exact rational shape parameters, including the symmetric case lam = 0
for lam in [Integer(0), Rational(1, 2), Integer(1), Rational(-3, 2)]:
    assert check(lam) == 0, lam
```

CLAIM c2: $\log g = \log f_1 + \log f_2$ implies $(\log g)'' = (\log f_1)'' + (\log f_2)''$, so a product of logconcave densities is logconcave.
```python
from sympy import *

x = symbols('x', real=True)
f1 = Function('f1', positive=True)(x)
f2 = Function('f2', positive=True)(x)

g = f1*f2
lhs = diff(expand_log(log(g), force=True), x, 2)
rhs = diff(log(f1), x, 2) + diff(log(f2), x, 2)
assert simplify(lhs - rhs) == 0

# concrete logconcave pair: both second derivatives of log are <= 0, and so is the sum
h1 = -x**2          # log f1
h2 = -Abs(x)**2/3   # log f2 (smooth here)
assert simplify(diff(h1, x, 2)) <= 0
assert simplify(diff(h2, x, 2)) <= 0
assert simplify(diff(h1 + h2, x, 2)) <= 0
```

CLAIM c3: For concave $h$, the interior chord is a lower bound and the two neighbouring extended chords are each an upper bound on $(x_i,x_{i+1})$.
```python
from sympy import *
from sympy.calculus.util import minimum

x = symbols('x', real=True)
h = -x**2                      # strictly concave, so h(x) = log f(x) for a logconcave f

nodes = [Integer(-2), Integer(-1), Integer(0), Integer(1)]   # x_0, x_1, x_2, x_3

def L(a, b):
    """straight line through (a, h(a)) and (b, h(b))"""
    ha, hb = h.subs(x, a), h.subs(x, b)
    return ha + (hb - ha)*(x - a)/(b - a)

x0, x1, x2, x3 = nodes
I = Interval(x1, x2)           # the interval (x_1, x_2) = (-1, 0)

lower = L(x1, x2)              # L_{i,i+1}
upA   = L(x0, x1)              # L_{i-1,i}
upB   = L(x2, x3)              # L_{i+1,i+2}

# chord lies below the concave function
assert minimum(h - lower, x, I) >= 0
# each extended neighbouring chord lies above the concave function
assert minimum(upA - h, x, I) >= 0
assert minimum(upB - h, x, I) >= 0
# hence the min of the two is also an upper bound
assert minimum(Min(upA, upB) - h, x, I) >= 0
```
