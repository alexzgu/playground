CLAIM c1: The Gaussian integral collapsing the two exponents equals $\sqrt{2\pi/(n/\sigma^2+1/\delta^2)}\,\exp\!\left(-\tfrac12\,\frac{(n/\sigma^2)(1/\delta^2)}{n/\sigma^2+1/\delta^2}(\bar{y}-\theta)^2\right)$.

```python
from sympy import *

mu, ybar, th = symbols('mu ybar theta', real=True)

# Symbolic in ybar, theta; exact Rational values for (n, sigma^2, delta^2)
# so the integral closes decisively.
for n, s2, d2 in [(2, Integer(1), Integer(3)),
                  (5, Integer(2), Integer(1)),
                  (3, Rational(1, 2), Rational(4, 3))]:
    a = Rational(n) / s2      # n / sigma^2
    b = 1 / d2                # 1 / delta^2
    lhs = integrate(exp(-a / 2 * (ybar - mu)**2 - b / 2 * (mu - th)**2),
                    (mu, -oo, oo))
    rhs = sqrt(2 * pi / (a + b)) * exp(-Rational(1, 2) * (a * b / (a + b)) * (ybar - th)**2)
    assert simplify(lhs - rhs) == 0, (n, s2, d2)
```
