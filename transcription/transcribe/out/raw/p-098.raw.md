CLAIM c1: the bin midpoint $(a_g + a_{g-1})/2$ lies strictly between $a_{g-1}$ and $a_g$.
```python
from sympy import *

a_lo, d = symbols('a_lo d', real=True, positive=True)
# a_{g-1} = a_lo, a_g = a_lo + d with d > 0, so a_{g-1} < a_g
a_prev = a_lo
a_g = a_lo + d

theta = (a_g + a_prev) / 2

# midpoint strictly inside the bin
assert simplify(theta - a_prev) == d/2
assert simplify(a_g - theta) == d/2
assert ask(Q.positive(simplify(theta - a_prev)))
assert ask(Q.positive(simplify(a_g - theta)))

# and it is exactly the average of the endpoints
assert simplify(2*theta - (a_g + a_prev)) == 0
```

CLAIM c2: the quadrature weights $w_g = g(\theta_g)\pi(\theta_g) / \sum_{h=1}^G g(\theta_h)\pi(\theta_h)$ sum to one.
```python
from sympy import *

G = 5
# arbitrary positive values for g(theta_g) * pi(theta_g)
vals = [Rational(3, 2), Rational(7, 5), Rational(1, 3), Rational(9, 4), Rational(2, 7)]
assert len(vals) == G

denom = sum(vals)
w = [v / denom for v in vals]

assert simplify(sum(w) - 1) == 0

# symbolic version for general G = 4
u = symbols('u1:5', positive=True)
denom_s = Add(*u)
w_s = [ui / denom_s for ui in u]
assert simplify(Add(*w_s) - 1) == 0
```

CLAIM c3: marginalizing $\pi(\boldsymbol{\mu}|\boldsymbol{y}) \approx \sum_g \left[\prod_{i=1}^{l}\pi(\mu_i|\theta_g,\boldsymbol{y})\right] w_g$ over $\mu_j$ for $j\neq i$ returns $\sum_g \pi(\mu_i|\theta_g,\boldsymbol{y}) w_g$.
```python
from sympy import *

# l = 3 units, G = 2 mixture components, each mu_i supported on {0, 1, 2}
l, G = 3, 2
support = [0, 1, 2]

# pi(mu_i | theta_g, y): rows indexed by (g, i), each a proper pmf over support
pmf = {
    (0, 0): [Rational(1, 2), Rational(1, 3), Rational(1, 6)],
    (0, 1): [Rational(1, 4), Rational(1, 4), Rational(1, 2)],
    (0, 2): [Rational(1, 5), Rational(3, 5), Rational(1, 5)],
    (1, 0): [Rational(1, 6), Rational(1, 2), Rational(1, 3)],
    (1, 1): [Rational(2, 5), Rational(2, 5), Rational(1, 5)],
    (1, 2): [Rational(1, 3), Rational(1, 3), Rational(1, 3)],
}
for row in pmf.values():
    assert sum(row) == 1

w = [Rational(3, 7), Rational(4, 7)]
assert sum(w) == 1

# joint mixture-of-products over (mu_0, mu_1, mu_2)
from itertools import product as iproduct
def joint(mu):
    return Add(*[w[g] * Mul(*[pmf[(g, i)][mu[i]] for i in range(l)]) for g in range(G)])

# it is a proper joint pmf
assert simplify(Add(*[joint(mu) for mu in iproduct(support, repeat=l)]) - 1) == 0

# marginalize out mu_j, j != i, for each i, and compare to the booklet's formula
for i in range(l):
    for val in support:
        marg = Add(*[joint(mu) for mu in iproduct(support, repeat=l) if mu[i] == val])
        booklet = Add(*[w[g] * pmf[(g, i)][val] for g in range(G)])
        assert simplify(marg - booklet) == 0
```
