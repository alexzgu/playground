CLAIM c1: For a finite measurable partition $A_1,\dots,A_k$ of $\mathcal{X}$, the Dirichlet parameters $\alpha\mathbb{G}_0(A_i)$ sum to $\alpha$, and the resulting Dirichlet marginal means are $E[\mathbb{G}(A_i)] = \mathbb{G}_0(A_i)$.
```python
from sympy import symbols, Rational, simplify, Symbol

alpha = Symbol('alpha', positive=True)

# A finite measurable partition induces a probability vector under G0:
# G0(A_1), ..., G0(A_k) are nonnegative and sum to 1.
# Take an exact rational partition measure as a concrete decisive instance.
G0 = [Rational(1, 2), Rational(1, 6), Rational(1, 4), Rational(1, 12)]
assert sum(G0) == 1

# Dirichlet parameters as the booklet writes them.
params = [alpha * g for g in G0]

# (a) total concentration is exactly alpha
assert simplify(sum(params) - alpha) == 0

# (b) Dirichlet mean of coordinate i is param_i / sum(params) = G0(A_i)
total = sum(params)
for g, p in zip(G0, params):
    assert simplify(p / total - g) == 0

# (c) the same holds for an arbitrary symbolic partition of size k=3
g1, g2 = symbols('g1 g2', positive=True)
g3 = 1 - g1 - g2                      # partition constraint
sym = [alpha * g1, alpha * g2, alpha * g3]
assert simplify(sum(sym) - alpha) == 0
assert simplify(sym[0] / sum(sym) - g1) == 0
assert simplify(sym[1] / sum(sym) - g2) == 0
assert simplify(sym[2] / sum(sym) - g3) == 0
```
