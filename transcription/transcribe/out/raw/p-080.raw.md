```python
from sympy import symbols, Rational, binomial, integrate, N

p = symbols('p', positive=True)

def bayes_factor(n, x, p0):
    num = binomial(n, x) * p0**x * (1 - p0)**(n - x)
    den = Rational(1, 1) / (1 - p0) * integrate(
        binomial(n, x) * p**x * (1 - p)**(n - x), (p, p0, 1))
    return num / den

p0 = Rational(3, 10)

de = bayes_factor(100, 37, p0)
dc = bayes_factor(97, 31, p0)

assert round(float(N(de, 30)), 2) == 2.00, float(N(de, 30))
assert round(float(N(dc, 30)), 2) == 8.10, float(N(dc, 30))
```
