from sympy import Matrix, symbols

x0, x1, x2, x3, x4 = symbols('x0 x1 x2 x3 x4', real=True)

from sympy import symbols, integrate, oo, simplify, Rational

alpha = symbols('alpha', positive=True)
p_alpha = 1 / (1 + alpha)**2

total = integrate(p_alpha, (alpha, 0, oo))
assert simplify(total - 1) == 0

# density is strictly positive on the support
assert simplify(p_alpha.subs(alpha, Rational(1, 2))) == Rational(4, 9)
