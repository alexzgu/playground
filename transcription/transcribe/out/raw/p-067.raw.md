CLAIM c1: The prior $\pi(\mu,\tau)=1/(1+\tau)^2$ on $(0,1)\times(0,\infty)$ is proper (integrates to 1).
```python
from sympy import *

mu, tau = symbols('mu tau', positive=True)
pi_joint = 1/(1+tau)**2

# integrate over tau in (0, oo), then mu in (0, 1)
inner = integrate(pi_joint, (tau, 0, oo))
total = integrate(inner, (mu, 0, 1))

assert simplify(inner - 1) == 0        # tau-marginal mass is finite
assert simplify(total - 1) == 0        # proper: total mass exactly 1
assert total.is_finite is True
```

CLAIM c2: The stated $\widehat{CPO}_r$ estimator is exactly the harmonic mean of the $M$ values $f(y_r\mid\theta^{(h)})$.
```python
from sympy import *

M = 5
a = symbols('a1:6', positive=True)     # a_h = f(y_r | theta^(h))

booklet = (Rational(1, M) * sum(1/ah for ah in a))**-1
harmonic_mean = M / sum(1/ah for ah in a)

assert simplify(booklet - harmonic_mean) == 0

# sanity: for equal values the harmonic mean is that common value
c = symbols('c', positive=True)
assert simplify(booklet.subs({ah: c for ah in a}) - c) == 0
```
