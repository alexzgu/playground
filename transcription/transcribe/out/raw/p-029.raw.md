===CLAIMS===

CLAIM c1: Poisson Fisher information is $1/\lambda$ and Jeffreys' prior $\lambda^{-1/2}$ is improper on $(0,\infty)$.
```python
from sympy import *

lam = symbols('lambda', positive=True)
x = symbols('x', integer=True, nonnegative=True)

logf = x*log(lam) - lam - log(factorial(x))
d1 = diff(logf, lam)
assert simplify(d1 - (x/lam - 1)) == 0

d2 = diff(logf, lam, 2)
assert simplify(d2 - (-x/lam**2)) == 0

# E[X] = lambda for Poisson: -E[d2] = lambda/lambda**2 = 1/lambda
info = simplify(-d2.subs(x, lam))
assert simplify(info - 1/lam) == 0

# Jeffreys prior lambda^{-1/2} is improper on (0, oo)
assert integrate(lam**Rational(-1, 2), (lam, 0, oo)) == oo
```

CLAIM c2: Binomial Fisher information equals $n[1/p + 1/(1-p)] = n/(p(1-p))$.
```python
from sympy import *

p = symbols('p', positive=True)
n = symbols('n', positive=True, integer=True)
xs = symbols('x', positive=True)

logf = log(binomial(n, xs)) + xs*log(p) + (n - xs)*log(1 - p)
d1 = diff(logf, p)
assert simplify(d1 - (xs/p - (n - xs)/(1 - p))) == 0

d2 = diff(logf, p, 2)
assert simplify(d2 - (-xs/p**2 - (n - xs)/(1 - p)**2)) == 0

# E[X] = n*p  =>  E[d2] = -n*p/p**2 - n*(1-p)/(1-p)**2
Ed2 = simplify(d2.subs(xs, n*p))
assert simplify(Ed2 - (-n*p/p**2 - n*(1 - p)/(1 - p)**2)) == 0

info = simplify(-Ed2)
assert simplify(info - n*(1/p + 1/(1 - p))) == 0
assert simplify(info - n/(p*(1 - p))) == 0
```

CLAIM c3: The Beta(1/2,1/2) density equals $1/(\pi\sqrt{p(1-p)})$ and is proper.
```python
from sympy import *

p = symbols('p', positive=True)

# B(1/2,1/2) = Gamma(1/2)^2 / Gamma(1) = pi
B = gamma(Rational(1, 2))*gamma(Rational(1, 2))/gamma(1)
assert simplify(B - pi) == 0
assert simplify(beta(Rational(1, 2), Rational(1, 2)).rewrite(gamma) - pi) == 0

# density p^{-1/2}(1-p)^{-1/2}/B(1/2,1/2) equals 1/(pi*sqrt(p(1-p))) on 0<p<1
q = symbols('q', positive=True)  # q = 1 - p, so both factors are explicitly positive
dens = p**(Rational(1, 2) - 1)*q**(Rational(1, 2) - 1)/B
target = 1/(pi*sqrt(p*q))
assert simplify(dens - target) == 0

# proper: integrates to 1 over (0,1)
I = integrate(1/(pi*sqrt(p*(1 - p))), (p, 0, 1))
assert simplify(I - 1) == 0

# numeric cross-check of propriety
from scipy.integrate import quad
import math
val, err = quad(lambda t: 1/(math.pi*math.sqrt(t*(1 - t))), 0, 1)
assert abs(val - 1) < 1e-8
```

CLAIM c4: For the normal mean with known variance, $I(\mu)=1/\sigma^2$, a constant.
```python
from sympy import *

mu, x = symbols('mu x', real=True)
s2 = symbols('sigma2', positive=True)

logf = -Rational(1, 2)*log(2*pi*s2) - (x - mu)**2/(2*s2)
d1 = diff(logf, mu)
assert simplify(d1 - (x - mu)/s2) == 0

d2 = diff(logf, mu, 2)
assert simplify(d2 + 1/s2) == 0

info = simplify(-d2)
assert simplify(info - 1/s2) == 0
assert diff(info, mu) == 0  # constant in mu
```

CLAIM c5: For the normal variance with known mean, $E[\partial^2_{(\sigma^2)}\log f]=-1/(2(\sigma^2)^2)$, $I(\sigma^2)=1/(2(\sigma^2)^2)$, and $\sqrt{I}=1/(\sqrt2\,\sigma^2)$.
```python
from sympy import *

mu, x = symbols('mu x', real=True)
v = symbols('v', positive=True)   # v = sigma^2

logf = -Rational(1, 2)*log(2*pi) - Rational(1, 2)*log(v) - (x - mu)**2/(2*v)
d1 = diff(logf, v)
assert simplify(d1 - (-1/(2*v) + (x - mu)**2/(2*v**2))) == 0

d2 = diff(logf, v, 2)
assert simplify(d2 - (1/(2*v**2) - (x - mu)**2/v**3)) == 0

# E[(X-mu)^2] = v  under N(mu, v); take expectation of d2 by integrating the density
dens = 1/sqrt(2*pi*v)*exp(-(x - mu)**2/(2*v))
Ed2 = simplify(integrate(d2*dens, (x, -oo, oo)))
assert simplify(Ed2 - (1/(2*v**2) - 1/v**2)) == 0
assert simplify(Ed2 + 1/(2*v**2)) == 0

info = simplify(-Ed2)
assert simplify(info - 1/(2*v**2)) == 0

# Jeffreys prior: sqrt(I) = 1/(sqrt(2)*v), proportional to 1/v
assert simplify(sqrt(info) - 1/(sqrt(2)*v)) == 0
assert simplify(sqrt(info)*v - 1/sqrt(2)) == 0   # ratio is a constant => proportional to 1/v
```
