CLAIM c1: Under the reference prior $p(\mu,\sigma^2)\propto\sigma^{-2}$, the marginal posterior of $t=(\mu-\bar x)/(s/\sqrt n)$ is Student-$t$ with $n-1$ df (checked exactly at $n=5$).
```python
from sympy import *

n = 5
t = symbols('t', real=True)
sig2 = symbols('sig2', positive=True)
s = symbols('s', positive=True)

# joint posterior kernel: p(mu,sig2|x) ∝ (sig2)^(-n/2 - 1) exp(-[(n-1)s^2 + n(mu-xbar)^2]/(2 sig2)).
# Substitute mu = xbar + t*s/sqrt(n) BEFORE integrating out sig2, so the exponent
# is manifestly negative: (n-1)s^2 + n(mu-xbar)^2 = s^2((n-1) + t^2).
A = s**2*((n - 1) + t**2)
kern_t = sig2**(-Rational(n, 2) - 1) * exp(-A/(2*sig2)) * s/sqrt(n)  # Jacobian d(mu)/dt included

dens_t = simplify(integrate(kern_t, (sig2, 0, oo)))

# Student-t_{n-1} density
tdens = gamma(Rational(n, 2))/(sqrt((n-1)*pi)*gamma(Rational(n-1, 2))) * (1 + t**2/(n-1))**(-Rational(n, 2))

# the two must agree up to a constant free of t: ratio is t-free
ratio = simplify(dens_t / tdens)
assert simplify(diff(ratio, t)) == 0

# and dens_t normalizes to the t_{n-1} law after dividing by its own integral
Z = integrate(dens_t, (t, -oo, oo))
assert simplify(dens_t/Z - tdens) == 0
```

CLAIM c4: Beta–Binomial conjugacy: with $p\sim\text{Beta}(\mu\tau,(1-\mu)\tau)$ and $x\mid p\sim\text{Bin}(n,p)$, the posterior at $x=s$ is $\text{Beta}(s+\mu\tau,\ n-s+(1-\mu)\tau)$.
```python
from sympy import *

p = symbols('p', positive=True)
# exact rational parameters so the integrals close
mu_, tau_ = Rational(3, 10), Integer(100)   # a = 30, b = 70
a, b = mu_*tau_, (1 - mu_)*tau_
n_, s_ = Integer(12), Integer(5)
Bfn = lambda x, y: gamma(x)*gamma(y)/gamma(x + y)   # Beta function, evaluates exactly

prior = p**(a - 1)*(1 - p)**(b - 1)/Bfn(a, b)
lik = binomial(n_, s_)*p**s_*(1 - p)**(n_ - s_)

post = simplify(prior*lik / integrate(prior*lik, (p, 0, 1)))

A, B = s_ + a, n_ - s_ + b
target = p**(A - 1)*(1 - p)**(B - 1)/Bfn(A, B)

assert simplify(post - target) == 0
```

CLAIM c7: 95% HPD intervals: $\text{Beta}(67,133)\to(0.270,0.401)$; $\text{Beta}(61,136)\to(0.246,0.375)$, to 3 dp.
```python
from scipy.stats import beta as bd
from scipy.optimize import brentq

def hpd(a, b, mass=0.95):
    # find lower-tail prob q such that density at the two endpoints matches
    def gap(q):
        lo = bd.ppf(q, a, b)
        hi = bd.ppf(q + mass, a, b)
        return bd.pdf(lo, a, b) - bd.pdf(hi, a, b)
    q = brentq(gap, 1e-12, 1 - mass - 1e-12, xtol=1e-15, rtol=1e-15)
    lo, hi = bd.ppf(q, a, b), bd.ppf(q + mass, a, b)
    # sanity: it really is a 95% interval and endpoint densities agree
    assert abs((bd.cdf(hi, a, b) - bd.cdf(lo, a, b)) - mass) < 1e-10
    assert abs(bd.pdf(lo, a, b) - bd.pdf(hi, a, b)) < 1e-8
    return round(float(lo), 3), round(float(hi), 3)

assert hpd(67, 133) == (0.270, 0.401), hpd(67, 133)
assert hpd(61, 136) == (0.246, 0.375), hpd(61, 136)
```
