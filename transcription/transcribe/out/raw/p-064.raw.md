CLAIM c1: The interior-segment integral of the piecewise-exponential envelope is $e^{\beta_i}(e^{\alpha_i x_{i+1}} - e^{\alpha_i x_i})/\alpha_i$.

b = symbols('beta', real=True)
xi, xj = symbols('x_i x_ip1', real=True)

# CDF of the density proportional to exp(a*x+b) on [xi, xj]
Z = integrate(exp(a*t + b), (t, xi, xj))
F = integrate(exp(a*t + b), (t, xi, x)) if False else None
