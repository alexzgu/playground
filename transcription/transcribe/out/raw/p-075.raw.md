CLAIM c2: Beta(5,500) 95% H.P.D. interval is (2.42e-3, 18.62e-3) to printed precision.
```python
from scipy.stats import beta
from scipy.optimize import minimize_scalar

d = beta(5, 500)

def width(a):
    return d.ppf(d.cdf(a) + 0.95) - a

res = minimize_scalar(width, bounds=(1e-12, d.ppf(0.05)), method="bounded",
                      options={"xatol": 1e-14})
lo = res.x
hi = d.ppf(d.cdf(lo) + 0.95)
# HPD endpoints must have equal density
assert abs(d.pdf(lo) - d.pdf(hi)) < 1e-4 * d.pdf(lo)
assert round(lo * 1e3, 2) == 2.42, round(lo * 1e3, 2)
assert round(hi * 1e3, 2) == 18.62, round(hi * 1e3, 2)
```

CLAIM c3: Poisson(n*lambda) likelihood with flat prior p(lambda)=1 yields posterior Gamma(x+1, rate n).
```python
from sympy import *

lam, n = symbols("lam n", positive=True)
x = symbols("x", positive=True, integer=True)

# likelihood times flat prior
kernel = exp(-n * lam) * (n * lam) ** x / factorial(x)
norm = integrate(kernel, (lam, 0, oo))
posterior = simplify(kernel / norm)

# Gamma(shape=x+1, rate=n) density
gamma_pdf = n ** (x + 1) * lam ** x * exp(-n * lam) / gamma(x + 1)

for xv in [0, 1, 5, 16]:
    diff = simplify(posterior.subs(x, xv) - gamma_pdf.subs(x, xv))
    assert diff == 0, (xv, diff)
```
