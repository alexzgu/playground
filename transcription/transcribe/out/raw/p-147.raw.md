CLAIM c1: The binomial expansion used to introduce the latent variable: $\sum_{z=0}^{x}\binom{x}{z}p^z q^{x-z} = (p+q)^x$.
```python
from sympy import *

p, q = symbols('p q', positive=True)
z = symbols('z', integer=True)

# check for a range of x; the identity is the binomial theorem
for xv in range(0, 8):
    lhs = Sum(binomial(xv, z) * p**z * q**(xv - z), (z, 0, xv)).doit()
    rhs = (p + q)**xv
    assert simplify(expand(lhs) - expand(rhs)) == 0, xv
```

CLAIM c2: Normalizing the augmented joint $\pi(p,q,z\mid x)\propto\binom{x}{z}p^zq^{x-z}p^{n-x}$ over $z=0,\dots,x$ yields $z\mid p,q,x\sim Binomial(x,\;p/(p+q))$.
```python
from sympy import *

p, q = symbols('p q', positive=True)
zz = symbols('zz', integer=True)

for nv in range(1, 7):
    for xv in range(0, nv + 1):
        # joint kernel in z (p^{n-x} is a constant in z but keep it explicit)
        kern = lambda zv: binomial(xv, zv) * p**zv * q**(xv - zv) * p**(nv - xv)
        norm = simplify(Sum(kern(zz), (zz, 0, xv)).doit())
        # normalizing constant must be p^{n-x} (p+q)^x
        assert simplify(norm - p**(nv - xv) * (p + q)**xv) == 0

        for zv in range(0, xv + 1):
            cond = simplify(kern(zv) / norm)
            theta = p / (p + q)
            binom_pmf = binomial(xv, zv) * theta**zv * (1 - theta)**(xv - zv)
            assert simplify(cond - binom_pmf) == 0, (nv, xv, zv)
```

CLAIM c3: Normalizing that joint over $p\in(0,1)$ yields $p\mid z,q,x\sim Beta(n-x+z+1,\,1)$.
```python
from sympy import *

p, q = symbols('p q', positive=True)

def beta_pdf(t, a, b):
    return t**(a - 1) * (1 - t)**(b - 1) / beta(a, b)

for nv in range(1, 7):
    for xv in range(0, nv + 1):
        for zv in range(0, xv + 1):
            kern = binomial(xv, zv) * p**zv * q**(xv - zv) * p**(nv - xv)
            norm = integrate(kern, (p, 0, 1))
            assert norm != 0
            cond = simplify(kern / norm)
            # booklet: Beta(n - x + z + 1, 1)
            claimed = beta_pdf(p, Integer(nv - xv + zv + 1), Integer(1))
            assert simplify(cond - claimed) == 0, (nv, xv, zv)
```
