CLAIM c1: The beta-binomial marginal integral equals $B(s+\mu\tau,\, n-s+(1-\mu)\tau)/B(\mu\tau,(1-\mu)\tau)$.
```python
from sympy import symbols, Rational, integrate, beta, gamma, simplify

p = symbols('p', positive=True)

# Exact rational parameters; a = mu*tau, b = (1-mu)*tau.
cases = [
    (5, 2, Rational(1, 3), 6),    # a = 2,     b = 4
    (2, 1, Rational(1, 2), 3),    # a = 3/2,   b = 3/2
    (4, 0, Rational(2, 5), 10),   # a = 4,     b = 6
    (6, 6, Rational(3, 4), 8),    # a = 6,     b = 2
    (3, 1, Rational(1, 6), Rational(9, 2)),  # a = 3/4, b = 15/4
]

for n, s, mu, tau in cases:
    a = mu * tau
    b = (1 - mu) * tau
    integrand = p**s * (1 - p)**(n - s) * p**(a - 1) * (1 - p)**(b - 1) / beta(a, b)
    lhs = integrate(integrand, (p, 0, 1))
    rhs = beta(s + a, n - s + b) / beta(a, b)
    # rewrite unevaluated beta(...) via gamma so simplify can close the identity
    assert simplify((lhs - rhs).rewrite(gamma)) == 0, (n, s, mu, tau, lhs, rhs)
```
