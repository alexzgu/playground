CLAIM c1: The ZOL posterior-mode denominator $s+\alpha+n+\beta-s-2$ equals $n+\alpha+\beta-2$; the printed right-hand side drops the $n$, so the stated equality fails whenever $n\neq 0$.
```python
from sympy import *

s, n, a, b = symbols('s n alpha beta', positive=True)

lhs_den = s + a + n + b - s - 2
# The left denominator is exactly n + alpha + beta - 2.
assert simplify(lhs_den - (n + a + b - 2)) == 0

lhs = (s + a - 1) / lhs_den
correct = (s + a - 1) / (n + a + b - 2)
printed = (s + a - 1) / (a + b - 2)

assert simplify(lhs - correct) == 0

# The printed RHS is NOT equal to the LHS: at a concrete admissible point
# (n=10, alpha=2, beta=3, s=4) with alpha>1 and alpha+beta>2, they differ.
vals = {n: Integer(10), a: Integer(2), b: Integer(3), s: Integer(4)}
assert lhs.subs(vals) == Rational(5, 13)
assert printed.subs(vals) == Rational(5, 3)
assert simplify(lhs - printed) != 0

# Also confirm `correct` really is the mode of Beta(s+alpha, n-s+beta),
# i.e. the stationary point of the log-density, for those values.
p = symbols('p', positive=True)
A, B = 4 + 2, 10 - 4 + 3          # s+alpha = 6, n-s+beta = 9
logf = (A - 1) * log(p) + (B - 1) * log(1 - p)
mode = solve(Eq(diff(logf, p), 0), p)
assert mode == [Rational(5, 13)]
```

CLAIM c2: Normal–normal posterior: $\mu\mid x\sim N[\lambda\bar{x}+(1-\lambda)\theta,\ (1-\lambda)\delta^2]$ with $\lambda=\delta^2/(\delta^2+\sigma^2/n)$ (note the booklet prints $(1-\lambda)\mu$ for the mean).
```python
from sympy import *

sigma, delta, n = symbols('sigma delta n', positive=True)
xbar, theta = symbols('xbar theta', real=True)

lam = delta**2 / (delta**2 + sigma**2 / n)

# Conjugate normal update: precisions add.
post_prec = n / sigma**2 + 1 / delta**2
post_var = 1 / post_prec
post_mean = (n * xbar / sigma**2 + theta / delta**2) / post_prec

assert simplify(post_mean - (lam * xbar + (1 - lam) * theta)) == 0
assert simplify(post_var - (1 - lam) * delta**2) == 0

# The booklet's printed mean uses mu in place of theta; these differ.
mu = symbols('mu', real=True)
assert simplify(post_mean - (lam * xbar + (1 - lam) * mu)) != 0

# Independent confirmation by completing the square in the exponent
# with small exact values: n=2, sigma=1, delta=1, theta=0, data mean xbar.
m = symbols('m', real=True)
y1, y2 = Rational(1), Rational(3)          # xbar = 2
expo = -(y1 - m)**2 / 2 - (y2 - m)**2 / 2 - m**2 / 2
# quadratic in m: -(3/2)(m - mean)^2 + const  => mean = 4/3, var = 1/3
q = Poly(expand(expo), m)
c2_, c1_, _ = q.all_coeffs()
mean_hat = -c1_ / (2 * c2_)
var_hat = -1 / (2 * c2_)
lam_v = lam.subs({delta: 1, sigma: 1, n: 2})
assert mean_hat == (lam_v * 2 + (1 - lam_v) * 0)
assert var_hat == (1 - lam_v) * 1**2
```

CLAIM c3: For $U\sim U(0,1)$, $P[F(y_{i-1}) < U \le F(y_i)] = F(y_i)-F(y_{i-1})$ whenever $0\le F(y_{i-1})\le F(y_i)\le 1$.
```python
from sympy import *
from sympy.stats import Uniform, P as Prob

u = Uniform('u', 0, 1)
Fa, Fb = Rational(1, 4), Rational(7, 10)     # F(y_{i-1}) <= F(y_i) in [0,1]
assert Prob((u > Fa) & (u <= Fb)) == Fb - Fa

# Symbolic version via the uniform CDF on [0,1]: P[a < U <= b] = b - a.
a, b = symbols('a b', positive=True)
integrand = Integer(1)
assert simplify(integrate(integrand, (symbols('t'), a, b)) - (b - a)) == 0

# Endpoint cases the booklet's scheme relies on: the intervals partition (0,1].
ps = [Rational(1, 6), Rational(1, 3), Rational(1, 2)]
cdf = [sum(ps[:k + 1]) for k in range(len(ps))]
assert cdf[-1] == 1
prev = 0
for k, c in enumerate(cdf):
    assert Prob((u > prev) & (u <= c)) == ps[k]
    prev = c
```

CLAIM c4: $\phi=1/(1+e^{\theta})$ maps $(-\infty,\infty)$ onto $(0,1)$; $\phi=1/(1+\theta)$ maps $(0,\infty)$ onto $(0,1)$. Both are strictly monotone (one-to-one).
```python
from sympy import *

th = symbols('theta', real=True)
phi1 = 1 / (1 + exp(th))
assert limit(phi1, th, -oo) == 1
assert limit(phi1, th, oo) == 0
# strictly decreasing => one-to-one onto (0,1)
# (compare by difference: simplify() may return e.g. -1/(4*cosh(theta/2)**2))
d1 = diff(phi1, th)
assert simplify(d1 - (-exp(th) / (1 + exp(th))**2)) == 0
# derivative is negative everywhere: -e^t/(1+e^t)^2 < 0 for all real t
assert ask(Q.negative(-exp(th) / (1 + exp(th))**2), Q.real(th))
assert phi1.subs(th, 0) == Rational(1, 2)

t = symbols('t', positive=True)
phi2 = 1 / (1 + t)
assert limit(phi2, t, 0, '+') == 1
assert limit(phi2, t, oo) == 0
assert simplify(diff(phi2, t) + 1 / (1 + t)**2) == 0
assert phi2.subs(t, 1) == Rational(1, 2)
```
