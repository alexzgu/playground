```python
from statistics import NormalDist

N = NormalDist()
theta, mu0, delta = 20.0, 22.0, 10.0

def mu_of(u):
    arg = u + (1.0 - u) * N.cdf((theta - mu0) / delta)
    return theta + delta * N.inv_cdf(arg)

# The booklet asserts this transform draws mu from the prior truncated to mu >= mu0,
# so every u in (0,1) must map into the support.

    u = k / 1000.0
    assert mu_of(u) >= mu0 - 1e-9, (u, mu_of(u))
```

CLAIM c3: Normal-normal conjugacy — posterior $\mathcal{N}(\lambda\bar{y}+(1-\lambda)\theta,\;(1-\lambda)\delta^2)$ with $\lambda=\delta^2/(\delta^2+\sigma^2/n)$.
```python
from sympy import symbols, simplify, diff, Rational, exp, integrate, oo, sqrt, pi

mu, ybar, theta = symbols('mu ybar theta', real=True)
sigma, delta = symbols('sigma delta', positive=True)
n = symbols('n', positive=True)

s2 = sigma**2 / n          # variance of ybar given mu
d2 = delta**2
lam = d2 / (d2 + s2)
m = lam * ybar + (1 - lam) * theta      # claimed posterior mean
v = (1 - lam) * d2                      # claimed posterior variance

# log-posterior kernel (up to terms free of mu) from likelihood * prior
E = (ybar - mu)**2 / s2 + (mu - theta)**2 / d2
# claimed kernel
C = (mu - m)**2 / v

# The two quadratics in mu must differ by a constant (free of mu):
assert simplify(diff(E - C, mu)) == 0

# and the claimed variance is the harmonic combination
assert simplify(v - (d2 * s2) / (d2 + s2)) == 0
# and the claimed mean is the precision-weighted average
assert simplify(m - (ybar / s2 + theta / d2) / (1 / s2 + 1 / d2)) == 0

# Exact numeric closure with small Rationals: posterior normalizes and has the stated mean.
subs = {sigma: 1, delta: 2, n: 3, ybar: Rational(1, 2), theta: Rational(1, 5)}
s2n, d2n = (sigma**2 / n).subs(subs), d2.subs(subs)
kernel = exp(-Rational(1, 2) * ((ybar - mu)**2 / s2n + (mu - theta)**2 / d2n).subs(subs))
Z = integrate(kernel, (mu, -oo, oo))
post_mean = simplify(integrate(mu * kernel, (mu, -oo, oo)) / Z)
assert simplify(post_mean - m.subs(subs)) == 0
post_var = simplify(integrate((mu - post_mean)**2 * kernel, (mu, -oo, oo)) / Z)
assert simplify(post_var - v.subs(subs)) == 0
```
