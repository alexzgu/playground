CLAIM c1: $1/\sqrt{p(1-p)} = p^{-1/2}(1-p)^{-1/2} = p^{1/2-1}(1-p)^{1/2-1}$ on $(0,1)$
```python
from sympy import *

p = symbols('p', positive=True)
lhs = 1/sqrt(p*(1-p))
mid = p**Rational(-1, 2) * (1-p)**Rational(-1, 2)
rhs = p**(Rational(1, 2)-1) * (1-p)**(Rational(1, 2)-1)

# restrict to 0<p<1 so that (1-p) is positive
for q in [Rational(1, 7), Rational(1, 3), Rational(1, 2), Rational(9, 10)]:
    assert simplify(lhs.subs(p, q) - mid.subs(p, q)) == 0
    assert simplify(mid.subs(p, q) - rhs.subs(p, q)) == 0

assert simplify(mid - rhs) == 0
```

CLAIM c4: $p^s(1-p)^{n-s}$ is the $Beta(s+1, n-s+1)$ kernel
```python
from sympy import *

p = symbols('p', positive=True)

# symbolic: the normalizing constant of the kernel is B(s+1, n-s+1)
s, n = symbols('s n', positive=True)
kernel = p**s * (1-p)**(n-s)
beta_pdf = p**((s+1)-1) * (1-p)**((n-s+1)-1) / beta(s+1, n-s+1)
assert simplify(kernel/beta_pdf - beta(s+1, n-s+1)) == 0

# decisive numeric check: kernel integrates to B(s+1, n-s+1) for exact rationals
for sv, nv in [(2, 5), (3, 3), (1, 4)]:
    I = integrate(kernel.subs({s: Integer(sv), n: Integer(nv)}), (p, 0, 1))
    target = beta(Integer(sv)+1, Integer(nv)-Integer(sv)+1).rewrite(gamma)
    assert simplify(I - target) == 0
    pdf = beta_pdf.subs({s: Integer(sv), n: Integer(nv)}).rewrite(gamma)
    assert simplify(integrate(pdf, (p, 0, 1)) - 1) == 0
```

CLAIM c5: Haldane posterior $p^s(1-p)^{n-s}/(p(1-p)) = p^{s-1}(1-p)^{n-s-1}$ is $Beta(s, n-s)$, proper iff $s>0$, $n-s>0$
```python
from sympy import *

p = symbols('p', positive=True)
s, n = symbols('s n', positive=True)

lhs = p**s * (1-p)**(n-s) / (p*(1-p))
rhs = p**(s-1) * (1-p)**(n-s-1)
assert simplify(lhs - rhs) == 0

# it is the Beta(s, n-s) kernel
beta_pdf = p**(s-1) * (1-p)**((n-s)-1) / beta(s, n-s)
assert simplify(rhs/beta_pdf - beta(s, n-s)) == 0

# proper when s>0 and n-s>0
for sv, nv in [(2, 5), (1, 2)]:
    I = integrate(rhs.subs({s: Integer(sv), n: Integer(nv)}), (p, 0, 1))
    assert I.is_finite and simplify(I - beta(Integer(sv), Integer(nv)-Integer(sv))) == 0

# improper when s = 0 (Haldane prior itself): integral diverges
div = integrate((1/(p*(1-p))), (p, Rational(0), Rational(1, 2)))
assert div is S.Infinity or div.is_infinite
```
