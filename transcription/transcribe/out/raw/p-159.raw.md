CLAIM c1: For 0 < p < 1, the sequence xi_k = (1-p)^k * p is strictly decreasing in k, and sum_{k=1}^inf xi_k = 1-p.
```python
from sympy import *

p, k = symbols('p k', positive=True)

xi = lambda kk: (1 - p)**kk * p

# strictly decreasing: xi_{k+1} - xi_k = -p^2 (1-p)^k < 0 for 0 < p < 1
diff_expr = simplify(xi(k + 1) - xi(k))
assert simplify(diff_expr - (-p**2 * (1 - p)**k)) == 0

# sign check on the closed form for 0 < p < 1, k >= 1: p^2 > 0 and (1-p)^k > 0
for pv in [Rational(1, 10), Rational(1, 2), Rational(9, 10)]:
    for kv in range(1, 8):
        assert diff_expr.subs({p: pv, k: kv}) < 0

# the sequence sums to 1 - p (a geometric tail starting at k = 1), not to 1
total = summation((1 - p)**k * p, (k, 1, oo))
assert simplify(total.subs(p, Rational(1, 3)) - Rational(2, 3)) == 0
assert simplify(total.subs(p, Rational(1, 4)) - Rational(3, 4)) == 0
```
