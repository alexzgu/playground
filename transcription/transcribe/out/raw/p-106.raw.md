### 10.2.2 Auto correlation

Let $U^{(1)}, U^{(2)}, ..., U^{(M)}$ be $M$ iterates for variable $U$ after initial convergence. The auto correlation coefficient of the sequence $\{U^{(i)}\}$ at lag $k$ is:

$$r_k = \frac{\sum_{i=1}^{N-k}(U^{(i)} - \bar{U})(U^{(i+k)} - \bar{U})}{\sum_{i=1}^{N}(U^{(i)} - \bar{U})^2}, \;\text{ where } \bar{U} = N^{-1}\sum_{i=1}^{N} U^{(i)},$$

CLAIM c1: With $\mu_i \mid \theta \sim \mathcal{N}(\theta, \delta^2)$ independent and a flat prior on $\theta$, the conditional posterior of $\theta$ is $\mathcal{N}(\bar\mu, \delta^2/\ell)$.
```python
from sympy import *

theta = symbols('theta', real=True)
d2 = symbols('delta2', positive=True)
L = 3
mus = symbols('mu1:4', real=True)
mubar = sum(mus) / L

# log-kernel of prod_i N(mu_i | theta, delta^2), as a function of theta
E_post = sum((m - theta)**2 for m in mus) / (2 * d2)
# log-kernel of the claimed N(mubar, delta^2 / ell)
E_claim = (theta - mubar)**2 / (2 * (d2 / L))

# The two log-kernels must differ by a constant free of theta
diff_expr = expand(E_post - E_claim)
assert simplify(diff(diff_expr, theta)) == 0
# and hence coefficients of theta^1, theta^2 vanish
p = Poly(diff_expr, theta)
assert simplify(p.coeff_monomial(theta)) == 0
assert simplify(p.coeff_monomial(theta**2)) == 0

# sanity: the claimed variance really is delta^2/ell, not delta^2 or ell*delta^2
E_wrong = (theta - mubar)**2 / (2 * d2)
assert simplify(diff(expand(E_post - E_wrong), theta)) != 0
```

CLAIM c2: Prior $\sigma^{-2}\sim$ Gamma$(a/2, b/2)$ with $y_{ij}\mid\mu_i,\sigma^2\sim\mathcal{N}(\mu_i,\sigma^2)$ gives full conditional Gamma$\{(a+\sum n_i)/2,\;(b+\sum\sum(y_{ij}-\mu_i)^2)/2\}$.
```python
from sympy import *

tau = symbols('tau', positive=True)          # tau = sigma^{-2}
a, b = symbols('a b', positive=True)

ns = [2, 3]                                  # ell = 2, n_1 = 2, n_2 = 3
L = len(ns)
mus = symbols('mu1:3', real=True)
ys = [[symbols(f'y{i+1}{j+1}', real=True) for j in range(ns[i])] for i in range(L)]

# Gamma(shape, rate) kernel: tau^(shape-1) exp(-rate*tau)
prior = tau**(a/2 - 1) * exp(-b*tau/2)

# N(mu_i, 1/tau) kernels: sqrt(tau) * exp(-tau*(y-mu)^2/2)
lik = 1
for i in range(L):
    for j in range(ns[i]):
        lik *= sqrt(tau) * exp(-tau * (ys[i][j] - mus[i])**2 / 2)

posterior = prior * lik

N = sum(ns)
S = sum((ys[i][j] - mus[i])**2 for i in range(L) for j in range(ns[i]))
claimed_shape = (a + N) / 2
claimed_rate = (b + S) / 2
claimed = tau**(claimed_shape - 1) * exp(-claimed_rate * tau)

assert simplify(posterior / claimed - 1) == 0

# decisive: the shape must be (a+sum n_i)/2, not e.g. (a+ell)/2
bad = tau**((a + L)/2 - 1) * exp(-claimed_rate * tau)
assert simplify(posterior / bad - 1) != 0
```

CLAIM c3: Prior $\delta^{-2}\sim$ Gamma$(c/2, d/2)$ with $\mu_i\mid\theta,\delta^2\sim\mathcal{N}(\theta,\delta^2)$ gives full conditional Gamma$\{(c+\ell)/2,\;(d+\sum(\mu_i-\theta)^2)/2\}$.
```python
from sympy import *

tau = symbols('tau', positive=True)          # tau = delta^{-2}
c, d = symbols('c d', positive=True)
theta = symbols('theta', real=True)

L = 3
mus = symbols('mu1:4', real=True)

prior = tau**(c/2 - 1) * exp(-d*tau/2)

lik = 1
for m in mus:
    lik *= sqrt(tau) * exp(-tau * (m - theta)**2 / 2)

posterior = prior * lik

S = sum((m - theta)**2 for m in mus)
claimed = tau**((c + L)/2 - 1) * exp(-(d + S)/2 * tau)

assert simplify(posterior / claimed - 1) == 0

# decisive: rate must include d + sum (mu_i - theta)^2, not d alone
bad = tau**((c + L)/2 - 1) * exp(-d/2 * tau)
assert simplify(posterior / bad - 1) != 0
```
