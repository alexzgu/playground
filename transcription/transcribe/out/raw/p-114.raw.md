CLAIM c1: The factors $(1/\delta^2)^{a/2+1}e^{-b/(2\delta^2)}$ and $(1/\sigma^2)^{a/2+1}e^{-b/(2\sigma^2)}$ are the kernels of $IG(a/2, b/2)$.
```python
from sympy import *

s2, a, b = symbols('s2 a b', positive=True)

# Inverse-Gamma(alpha, beta) density: beta**alpha/gamma(alpha) * x**(-alpha-1) * exp(-beta/x)
def ig_density(x, alpha, beta):
    return beta**alpha / gamma(alpha) * x**(-alpha - 1) * exp(-beta / x)

booklet = (1/s2)**(a/2 + 1) * exp(-b/(2*s2))
dens = ig_density(s2, a/2, b/2)

# booklet factor must equal the IG(a/2, b/2) density up to a constant free of s2
ratio = simplify(dens / booklet)
assert simplify(diff(ratio, s2)) == 0
# and the constant is exactly the normalizing constant
assert simplify(ratio - (b/2)**(a/2)/gamma(a/2)) == 0
```

CLAIM c2: The $\beta$ full conditional is Normal with mean $(\sum\sum dd')^{-1}\sum\sum d(y-\nu)$ and covariance $\sigma^2(\sum\sum dd')^{-1}$.
```python
from sympy import *

s2 = symbols('s2', positive=True)
b0, b1 = symbols('b0 b1', real=True)
beta = Matrix([b0, b1])

# small exact design: ell=2 groups, n_i=2 each
data = [(1, Rational(1), Rational(3), Rational(1,2)),   # (i, x, y, nu_i)
        (1, Rational(2), Rational(5), Rational(1,2)),
        (2, Rational(4), Rational(2), Rational(-1,3)),
        (2, Rational(7), Rational(9), Rational(-1,3))]

# log of the beta-dependent part of the joint posterior
logk = 0
for (_, x, y, nu) in data:
    logk += -(y - (b0 + b1*x + nu))**2 / (2*s2)

D = zeros(2, 2)
v = zeros(2, 1)
for (_, x, y, nu) in data:
    d = Matrix([1, x])
    D += d * d.T
    v += d * (y - nu)

mean = D.inv() * v
cov = s2 * D.inv()
prec = cov.inv()

# Normal(mean, cov) log-kernel
diffv = beta - mean
logn = -(diffv.T * prec * diffv)[0, 0] / 2

# difference must be free of b0, b1  =>  same distribution
assert simplify(diff(logk - logn, b0)) == 0
assert simplify(diff(logk - logn, b1)) == 0
```

CLAIM c3: The $\nu_i$ full conditional is Normal with precision $1/\delta^2 + n_i/\sigma^2$ and mean $\frac{\sum_j (y_{ij}-\beta_0-\beta_1 x_{ij})/\sigma^2}{1/\delta^2 + n_i/\sigma^2}$.
```python
from sympy import *

s2, d2 = symbols('s2 d2', positive=True)
nu = symbols('nu', real=True)
n_i = 3
r = symbols('r1 r2 r3', real=True)   # r_ij = y_ij - b0 - b1*x_ij

# nu_i-dependent part of the log joint posterior
logk = -sum((ri - nu)**2 for ri in r)/(2*s2) - nu**2/(2*d2)

prec = 1/d2 + Rational(n_i)/s2
mean = (sum(r)/s2) / prec
logn = -prec*(nu - mean)**2 / 2       # Normal(mean, 1/prec) log-kernel

# must agree up to a constant free of nu
assert simplify(diff(logk - logn, nu)) == 0
# and the booklet's stated variance is 1/prec
assert simplify(1/prec - 1/(1/d2 + Rational(n_i)/s2)) == 0
```

CLAIM c4: The $\sigma^2$ full conditional is $IGamma\!\left(\frac{a+\sum_i n_i}{2},\frac{b+SSE}{2}\right)$.

from sympy import *

s2, a, b, N, SSE = symbols('s2 a b N SSE', positive=True)

# sigma^2-dependent factors of the joint posterior:
#   (1/sqrt(2*pi*s2))**N * exp(-SSE/(2*s2))  *  (1/s2)**(a/2+1) * exp(-b/(2*s2))
joint = s2**(-N/2) * exp(-SSE/(2*s2)) * (1/s2)**(a/2 + 1) * exp(-b/(2*s2))

alpha = (a + N)/2
beta_ = (b + SSE)/2
ig_kernel = s2**(-alpha - 1) * exp(-beta_/s2)

assert simplify(joint / ig_kernel - 1) == 0

CLAIM c6: $\hat{\beta}=[\sum\sum d_{ij}d_{ij}']^{-1}[\sum\sum d_{ij} y_{ij}]$ minimizes $\sum\sum (y_{ij}-d_{ij}'\beta)^2$.
```python
from sympy import *

b0, b1 = symbols('b0 b1', real=True)
beta = Matrix([b0, b1])

pts = [(Rational(1), Rational(3)), (Rational(2), Rational(5)),
       (Rational(4), Rational(2)), (Rational(7), Rational(9))]

SSE = sum((y - (b0 + b1*x))**2 for (x, y) in pts)

D = zeros(2, 2)
v = zeros(2, 1)
for (x, y) in pts:
    d = Matrix([1, x])
    D += d * d.T
    v += d * y
bhat = D.inv() * v

# stationary point of SSE
sol = solve([diff(SSE, b0), diff(SSE, b1)], [b0, b1], dict=True)[0]
assert simplify(bhat[0] - sol[b0]) == 0
assert simplify(bhat[1] - sol[b1]) == 0

# and it is a minimum: Hessian = 2*D is positive definite
H = Matrix(2, 2, lambda i, j: diff(SSE, [b0, b1][i], [b0, b1][j]))
assert simplify(H - 2*D) == zeros(2, 2)
assert all(ev > 0 for ev in H.eigenvals())
```
