```python
b0, b1 = symbols('beta0 beta1', real=True)
x1, x2, x3 = symbols('x1 x2 x3', real=True)

X = Matrix([[1, x1], [1, x2], [1, x3]])
beta = Matrix([b0, b1])

# substitute a design with full column rank so (X'X) is invertible
sub = {x1: 0, x2: 1, x3: 3}
Xs = X.subs(sub)
```

```python
from sympy import symbols, Matrix, eye, simplify, Rational

s2 = symbols('sigma2', positive=True)

# n = 5 design points, p = 2 parameters
xs = [Rational(0), Rational(1), Rational(2), Rational(4), Rational(7)]
n, p = len(xs), 2
X = Matrix([[1, xi] for xi in xs])
H = X * (X.T * X).inv() * X.T

M = eye(n) - H
# E[RSS] = E[y'(I-H)y] = sigma^2 * tr(I-H)   (the mean term vanishes since (I-H)X = 0)

assert simplify(M.trace() - (n - p)) == 0

E_S2 = simplify(s2 * M.trace() / (n - 2))
assert simplify(E_S2 - s2) == 0
```
