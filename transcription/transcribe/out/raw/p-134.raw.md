CLAIM c1: The printed $\sigma^2$ prior factor $(b/2)^{a/2}(1/\sigma^2)^{a/2+1}e^{-b/(2\sigma^2)}/\Gamma(a/2)$ integrates to 1 over $\sigma^2 \in (0,\infty)$.
```python
from sympy import *

s = symbols('s', positive=True)  # s = sigma^2

# Check for several exact rational (a, b) pairs; each must integrate to exactly 1.
for a_val, b_val in [(Rational(3), Rational(5)),
                     (Rational(1), Rational(2)),
                     (Rational(7, 2), Rational(1, 3)),
                     (Rational(4), Rational(4))]:
    dens = (b_val / 2)**(a_val / 2) * (1 / s)**(a_val / 2 + 1) \
           * exp(-b_val / (2 * s)) / gamma(a_val / 2)
    total = integrate(dens, (s, 0, oo))
    assert simplify(total - 1) == 0, (a_val, b_val, total)

# And confirm it coincides with the inverse-gamma(a/2, b/2) density symbolically.
a, b = symbols('a b', positive=True)
printed = (b / 2)**(a / 2) * (1 / s)**(a / 2 + 1) * exp(-b / (2 * s)) / gamma(a / 2)
alpha, beta = a / 2, b / 2
invgamma = beta**alpha / gamma(alpha) * s**(-alpha - 1) * exp(-beta / s)
assert simplify(printed - invgamma) == 0
```
